001    package org.hackystat.utilities.time.interval;
002    
003    import java.util.Iterator;
004    import java.util.NoSuchElementException;
005    
006    import org.hackystat.utilities.time.period.Month;
007    
008    /**
009     * Defines iterator for month interval.
010     *  
011     * @author Hongbing Kou
012     * @version $Id: MonthIterator.java,v 1.1.1.1 2005/10/20 23:56:40 johnson Exp $
013     */
014    public class MonthIterator implements Iterator<Month> {
015      /** Start month. */
016      private Month startMonth;
017      /** End month. */
018      private Month endMonth;
019      /** Current month. */
020      private Month currentMonth;
021       
022      /**
023       * Creates an iterator.
024       * 
025       * @param monthInterval Month Interval.
026       */
027      MonthIterator(MonthInterval monthInterval) {
028        this.startMonth = monthInterval.getStartMonth();
029        this.endMonth = monthInterval.getEndMonth();
030        this.currentMonth = this.startMonth.dec();
031      }
032      
033      /**
034       * Gets the next month.
035       * 
036       * @return Next month.
037       */  
038      public Month next() {
039        this.currentMonth = this.currentMonth.inc();
040        
041        if (this.currentMonth.compareTo(this.endMonth) > 0) {
042          throw new NoSuchElementException("Reaches the end of month interval already.");
043        }
044        
045        return this.currentMonth;     
046      }
047      
048      /**
049       * If it is before the end month it will be a good one.
050       * 
051       * @return Run our of interval or not.
052       */
053      public boolean hasNext() {
054        return this.currentMonth.compareTo(this.endMonth) < 0;
055      }
056      
057      /**
058       * Required for iterator().  It will throw UnSupportedMethodException.
059       */
060      public void remove() {
061        throw new UnsupportedOperationException("remove() is not supported by month iterator.");
062      }
063    }
