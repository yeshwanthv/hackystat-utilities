018      /**
019       * Test the APIs of day object.
020       * @throws Exception If problems occur.
021       */
022      public void testDay() throws Exception {
023        Day day = Day.getInstance("01-Jan-2001");
024        Day sameDay = Day.getInstance(2001, 0, 1);
025        assertEquals("Check the year string", "2001", day.getYearString());
026        assertEquals("Check the month string", "01", day.getMonthString());
027        assertEquals("Check the day string", "01", day.getDayString());
028        assertTrue("Checking the hashcode of the day object", day.hashCode() > 0);
029        assertEquals("Checking day and sameday.", day, sameDay);
030    
031        Day day2 = Day.getInstance();
032        Day day3 = Day.getInstance();
033        assertEquals("Checking that two days in same day are equal", day2, day3);
034        assertSame("Checking that two days in same day are identical", day2, day3);
035    
036        Day day4 = day3.inc(1);
037        assertTrue("Checking the comparison less", day3.compareTo(day4) < 0);
038        assertEquals("Checking the comparison equal", day3.compareTo(day3), 0);
039        assertTrue("Checking the comparison more", day4.compareTo(day3) > 0);
040        
041        assertTrue("Checking isBefore", day3.isBefore(day4));
042        assertFalse("Checking isBefore with the same day", day3.isBefore(day3));
043    
044        assertEquals("Checking daysBetween on same day", 0, Day.daysBetween(day, day));
045        assertEquals("Checking daysBetween on adjacent days.", 1, Day.daysBetween(day3, day4));
046        assertEquals("Checking daysBetween on adjacent days.", -1, Day.daysBetween(day4, day3));
047        assertEquals("Checking daysBetween on dist days.", 100, Day.daysBetween(day3, day3.inc(100)));
048        
049        TimeZone defaultTimeZone = TimeZone.getDefault();
050        Calendar cal = Calendar.getInstance(Locale.US);
051        // Test the DayCache because of daylight saving
052        if (defaultTimeZone.inDaylightTime(cal.getTime())) {
053          // January first is not in daylight saving
054          cal.set(Calendar.MONTH, 0);
055          cal.set(Calendar.DAY_OF_MONTH, 1);
056          
057          // 11:15pm
058          cal.set(Calendar.HOUR_OF_DAY, 23);           
059          cal.set(Calendar.MINUTE, 15);      
060          Date lateNight = cal.getTime();
061          Day.getInstance(lateNight);  // late night day. 
062          
063          // 9:30am
064          cal.set(Calendar.HOUR_OF_DAY, 9);           
065          cal.set(Calendar.MINUTE, 30);
066          Date officeHour = cal.getTime();
067    
068          Day officeHourDay = Day.getInstance(officeHour);
069          
070          // Gets next day
071          Day nextDay = officeHourDay.inc(1);
072          
073          assertNotSame("Next day should be 1/2 but 1/1 ", officeHourDay.compareTo(nextDay), 0);
074        }
075        else {
076          // July 4th is in daylight saving
077          cal.add(Calendar.YEAR, 1);
078          cal.set(Calendar.MONTH, 6);
079          cal.set(Calendar.DAY_OF_MONTH, 4);
080          
081          // 12:10am
082          cal.set(Calendar.HOUR_OF_DAY, 0);           
083          cal.set(Calendar.MINUTE, 10);
084          Date earlyMorning = cal.getTime();
085          Day.getInstance(earlyMorning); // early morning day
086          
087          // 9:30am
088          cal.set(Calendar.HOUR_OF_DAY, 9);           
089          cal.set(Calendar.MINUTE, 30);
090          Date officeHour = cal.getTime();
091          Day officeHourDay = Day.getInstance(officeHour);
092          
093          Day previousDay = officeHourDay.inc(-1);
094          assertNotSame("Previous day should be 7/3 not 7/4", officeHourDay.compareTo(previousDay), 0);
095        }
096      }

098      /** 
099       * Tests the day light savings issue associated with getting a day.  When a Day instance is 
100       * created it is retrieved from DayCache. The DayCache must account for differences
101       * in timestamps due to TimeZone and Daylight Saving Issues. This unit tests ensures
102       * that we are able to correctly create Day objects for a 3 year period at 30 minute intervals.
103       * <p>
104       * A failure in this method is caused by a date (ie. 0-31) of the Day object does not
105       * equal the date of the Date object.  This indicates that the DayCache is not interpreting
106       * the TimeZone and DST correctly.
107       */
108      public void testDaylightSavingsIssues() {
109        Calendar calendar = Calendar.getInstance(Locale.US);
110        Calendar tempCal = Calendar.getInstance(Locale.US);
111        Calendar calendarNextYear = Calendar.getInstance(Locale.US);
112        tempCal.set(Calendar.YEAR, calendar.get(Calendar.YEAR) - 1);
113        calendarNextYear.set(Calendar.YEAR, calendar.get(Calendar.YEAR) + 2);
114        while (!tempCal.equals(calendarNextYear)) {
115          Day day = Day.getInstance(tempCal.getTime());
116          assertEquals("checking that the day values are the same " +
117              "[date=" + tempCal.getTime() + ", day=" + day + "]", 
118              tempCal.get(Calendar.DATE), Integer.parseInt(day.getDayString()));
119          tempCal.add(Calendar.MINUTE, 30);
120        }
121      }

123      /**
124       * Tests <code>getFirstTickOfTheDay</code> and <code>getLastTickOfTheDay</code> method.
125       */
126      public void testGetFirstTickOfTheDay() {
127        long millisInADay = 24 * 60 * 60 * 1000;
128        Calendar cal = Calendar.getInstance(Locale.US);
129        cal.set(2005, 0, 1, 10, 10, 10);
130        Day day = Day.getInstance(cal.getTime());
131        
132        cal.set(2005, 0, 1, 0, 0, 0);
133        cal.set(Calendar.MILLISECOND, 0);
134        long firstTickOfDay = cal.getTimeInMillis();
135        long lastTickOfDay = firstTickOfDay + millisInADay - 1;
136        assertEquals("Checking first tick", firstTickOfDay, day.getFirstTickOfTheDay());
137        assertEquals("Checking last tick", lastTickOfDay, day.getLastTickOfTheDay());
138      }
