001    package org.hackystat.utilities.time.period;
002    
003    import java.util.Calendar;
004    import java.util.Date;
005    import java.util.Locale;
006    import java.util.TimeZone;
007    
008    /**
009     * Provides a cache mechanism for obtaining a single "canonical" Day instance associated with any
010     * given UTC time stamp.  "Canonical" means that for any UTC value within a given day, the
011     * same Day instance will be returned.  Performance monitoring has indicated that certain types of
012     * analyses spend significant resources producing and discarding Day instances.  This class is
013     * designed to lower the overhead of such computations by maintaining a cache of Day instances
014     * that can be shared.  Note that the DayCache cannot be used in situations where the original UTC
015     * value is needed later, since the Day instance that is returned will not return the original UTC
016     * value through invocation of the getTime() method.
017     * <P>
018     * This class is a "helper" class to Day. It is used only within the constructors for Day. There
019     * are no other clients to this class. Normal users should manipulate Day instances by calling the
020     * associated Day.getInstance() methods.
021     * <p>
022     * DayCache implements a sliding "window" cache of 730 instances corresponding to 730 days.  The
023     * middle instance corresponds to the day that this cache was first used (which normally
024     * corresponds to the server startup day.  If the DayCache is asked for a Day instance outside
025     * this window, then a  non-cached, regular Day instance is returned. The reason for this design
026     * is to minimize object creation: we can do arithmetic to get the array index corresponding to a
027     * UTC value.
028     * <p>
029     * Use only Day.equals() to compare two Day instances. Do not use '=='.
030     * 
031     * The Calendar is forced to Locale.US to ensure constant week boundaries. 
032     * 
033     * @author Philip M. Johnson
034     * @version $Id: DayCache.java,v 1.1.1.1 2005/10/20 23:56:44 johnson Exp $
035     */
036    public class DayCache {
037      /** The singleton DayCache instance. */
038      private static DayCache theCache = new DayCache();
039    
040      /** Number of milliseconds in a day. */
041      private long millisInDay = 1000 * 60 * 60 * 24;
042    
043      /** The first UTC time that will be stored in the cache, initialized in the constructor. */
044      private long minUTC = 0;
045    
046      /** The last UTC time that will be stored in the cache, initialized in the constructor. */
047      private long maxUTC = 0;
048    
049      /** The array of cached Day instances. day[365] is the day the server started. */
050      private Day[] days = new Day[730];
051      
052      /** The date the DayCache was constructed. */
053      private Date startupDate;
054    
055      /**
056       * The singleton DayCache instance that provides access to a "window" of cached,  canonical Day
057       * instances.  The cache is lazily instantiated.
058       */
059      private DayCache() {
060        Calendar cal = Calendar.getInstance(Locale.US);
061        cal.set(Calendar.HOUR_OF_DAY, 0);
062        cal.set(Calendar.MINUTE, 0);
063        cal.set(Calendar.SECOND, 0);
064        cal.set(Calendar.MILLISECOND, 0);
065        // I use 364 to avoid off by one errors at expense of thinking hard. :-)
066        this.minUTC = cal.getTimeInMillis() - (364 * millisInDay);
067        this.maxUTC = cal.getTimeInMillis() + (364 * millisInDay);
068        
069        this.startupDate = new Date();
070      }
071    
072      /**
073       * Returns the singleton DayCache.
074       * 
075       * <p>
076       * To be used only by the Day class constructors.
077       * </p>
078       *
079       * @return The DayCache.
080       */
081      static DayCache getInstance() {
082        return theCache;
083      }
084    
085      /**
086       * Return the cached Day instance corresponding to timestamp if plus/minus 364 days from the day
087       * the Cache was initialized.  Otherwise, returns a new Day instance.  Note that cached Day
088       * instances will usually have an internal UTC value that is different  from timestamp.
089       * 
090       * <p>
091       * To be used only by the Day class constructors.
092       * </p>
093       *
094       * @param timestamp The UTC value falling within the Day of interest.
095       *
096       * @return The cached Day, or a new Day if it returns outside the cache.
097       */
098      Day getDay(long timestamp) {
099        if ((timestamp >= maxUTC) || (timestamp <= minUTC)) {
100          return new Day(timestamp);
101        }
102        else {
103          // the dstSavings value is used to correct the timestamps that have added the Daylight
104          // Savings Time to the UTC. the DST is usually configured in the OS by the user. 
105          // if the system clock isn't DST enabled, dstSavings will be zero.
106          int dstSavings = 0;
107          if (TimeZone.getDefault().inDaylightTime(this.startupDate) && 
108              !TimeZone.getDefault().inDaylightTime(new Date(timestamp))) {
109            dstSavings = -TimeZone.getDefault().getDSTSavings();
110          }
111          else if (!TimeZone.getDefault().inDaylightTime(this.startupDate) && 
112              TimeZone.getDefault().inDaylightTime(new Date(timestamp))) {
113            dstSavings = TimeZone.getDefault().getDSTSavings();
114          }
115          int index = (int) (((timestamp + dstSavings) - minUTC ) / millisInDay);
116          if (days[index] == null) {
117            days[index] = new Day(timestamp);
118          }
119    
120          return days[index];
121        }
122      }
123    }
