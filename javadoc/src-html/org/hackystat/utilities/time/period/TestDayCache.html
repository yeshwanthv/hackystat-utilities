001    package org.hackystat.utilities.time.period;
002    
003    import java.text.SimpleDateFormat;
004    import java.util.Calendar;
005    import java.util.Date;
006    import java.util.Locale;
007    
008    import junit.framework.TestCase;
009    
010    /**
011     * Tests the DayCache implementation, and provides a main() method for simple 
012     * performance evaluation.
013     *
014     * @author Philip M. Johnson
015     */
016    public class TestDayCache extends TestCase {
017      
018      /**
019       * Gets the passed date in yyyy-MM-dd format.
020       * @param date The date. 
021       * @return Today's date. 
022       */
023      private String getTestDate(Date date) {
024        SimpleDateFormat dayFormat = new SimpleDateFormat("yyyy-MM-dd", Locale.US);
025        return dayFormat.format(date);
026      }
027      
028      /**
029       * Returns the long value corresponding to the passed time for today.
030       *
031       * @param hourOfDay Hour of day.
032       * @param minute Minute of day.
033       * @param second Second of day.
034       * @param milli Millisecond of day.
035       *
036       * @return long value for today at specified time.
037       */
038      private long getTodayAt(int hourOfDay, int minute, int second, int milli) {
039        Calendar cal = Calendar.getInstance(Locale.US);
040        cal.set(Calendar.HOUR_OF_DAY, hourOfDay);
041        cal.set(Calendar.MINUTE, minute);
042        cal.set(Calendar.SECOND, second);
043        cal.set(Calendar.MILLISECOND, milli);
044        return cal.getTime().getTime();
045      }
046    
047      /**
048       * Tests to see that DayCache returns the same Day object for various times in the same day.
049       */
050      public void testUTCDayValue() {
051        DayCache dayCache = DayCache.getInstance();
052        Date today = new Date();
053        String todayString = getTestDate(today);
054    
055        assertEquals("Test that cached today is equals to an uncached today.",
056          dayCache.getDay(new Date().getTime()).getSimpleDayString(), todayString);
057        assertEquals("Test that the cached Date of 11:59:59.999 today returns a cached Today.",
058          dayCache.getDay(getTodayAt(23, 59, 59, 999)).getSimpleDayString(), todayString);
059        assertEquals("Test that the cached Date of 00:00:00.000 today returns a cached Today.",
060          dayCache.getDay(getTodayAt(0, 0, 0, 0)).getSimpleDayString(), todayString);
061        assertFalse("Test that the cached Date of tomorrow is not Today.",
062          dayCache.getDay(getTodayAt(23, 59, 59, 999) + 1).equals(dayCache.getDay(today.getTime())));
063        assertFalse("Test that the cached Date of yesterday is not Today.",
064          dayCache.getDay(getTodayAt(0, 0, 0, 0) - 1).equals(dayCache.getDay(today.getTime())));
065    
066        long twoYearUTC = today.getTime() + (1000 * 60 * 60 * 24 * 365 * 2);
067        assertEquals("Test that a day two years in the future returns a day instance",
068          dayCache.getDay(today.getTime() + twoYearUTC), new Day(today.getTime() + twoYearUTC));
069        assertEquals("Test that a day two years in the past returns a day instance",
070          dayCache.getDay(today.getTime() - twoYearUTC), new Day(today.getTime() - twoYearUTC));
071      }
072    
073      /**
074       * Supports performance analysis testing of DayCache. If this main is invoked with "-cache", it
075       * invokes the DayCache on times starting now and incrementing by 1 second for the next 364
076       * days.  Otherwise, does the same thing but just creates a new Day instance each time.
077       * 
078       * <p>
079       * Initial evaluation indicates that without the cache, it takes around 80 seconds to create a
080       * day instance for each second for 100 days. With the cache, it takes around 3 seconds.
081       * </p>
082       *
083       * @param args An arg of "-cache" means use the DayCache.
084       */
085      public static void main(String[] args) {
086        boolean useCache = (args.length > 0) && (args[0].equals("-cache"));
087        System.out.println("Starting TestDayCache with cache " + ((useCache) ? "enabled" : "disabled"));
088    
089        Date startTime = new Date();
090        long today = new Date().getTime();
091        for (int day = 1; day < 100; day++) {
092          for (int hour = 0; hour < 24; hour++) {
093            for (int minute = 0; minute < 60; minute++) {
094              for (int second = 0; second < 60; second++) {
095                long tstamp = today + (1000 * second * minute * hour * day);
096                if (useCache) {
097                  Day.getInstance(tstamp); 
098                }
099                else {
100                  new Day(tstamp);
101                }
102              }
103            }
104          }
105        }
106    
107        System.out.println("Start time: " + startTime);
108        System.out.println("End time: " + new Date());
109      }
110    }
