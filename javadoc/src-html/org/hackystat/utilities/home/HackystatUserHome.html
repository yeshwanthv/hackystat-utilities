001    package org.hackystat.utilities.home;
002    
003    import java.io.File;
004    
005    /**
006     * Provides a utility that returns the desired location of the dot-hackystat directory. 
007     * This defaults to the user.home System Property, but can be overridden by the user
008     * by providing the property hackystat.user.home.
009     * @author Philip Johnson
010     *
011     */
012    public class HackystatUserHome {
013      
014      /**
015       * Return a File instance representing the desired location of the .hackystat directory.
016       * Note that this directory may or may not exist. 
017       * @return A File instance representing the desired user.home directory. 
018       */
019      public static File getHome() {
020        String userHome = System.getProperty("hackystat.user.home", System.getProperty("user.home"));
021        return new File(userHome);
022      }
023    
024    }
