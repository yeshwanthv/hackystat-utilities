011    /**
012     * Validate syntax of email addresses. Does not probe to see if mailserver exists in DNS or online.
013     * See MailProber for that. See ValidateEmailFile for an example of how to use this class.
014     * 
015     * Minor reformatting.  Found in:
016     * http://www.velocityreviews.com/forums/
017     *  t128486-re-can-javamail-detect-a-nonexistant-email-address.html
018     * 
019     * @author Roedy Green, Canadian Mind Products
020     * @author Philip Johnson. 
021     * @version 1.0 to do: check validity of & in first part of email address. Appears in practice.
022     */
023    public class ValidateEmailSyntax {
024      
025      /**
026       * Returns true if the email appears to be valid.  
027       * @param email The email address of interest. 
028       * @return True if it appears to be valid. 
029       */
030      public static boolean isValid(String email) {
031        return howValid(email) >= 2;
032      }
033    
034      /**
035       * Check how likely an email address is to be valid. The higher the number returned, the more
036       * likely the address is valid. This method does not probe the internet in any way to see if the
037       * corresponding mail server or domain exists.
038       * 
039       * @param email bare computer email address. e.g. No "Roedy Green" <> style addresses. No local
040       *        addresses, eg. roedy.
041       * 
042       * @return 0 = email address is definitely malformed, e.g. missing @. ends in .invalid <br>
043       *         1 = address does not meet one of the valid patterns below. It still might be ok
044       *         according to some obscure rule in RFC 822 Java InternetAddress accepts it as valid.
045       *         <br>
046       *         2 = unknown top level domain. <br>
047       *         3 = dots at beginning or end, doubled in name. <br>
048       *         4 = address of form xxx@[209.139.205.2] using IP <br>
049       *         5 = address of form Dots _ or - in first part of name <br>
050       *         6 = addreess of form rare, but known, domain <br>
051       *         7 = address of form or any national suffix. <br>
052       *         8 = address of form the matching this national suffix, e.g. .ca in Canada, .de in
053       *         Germany <br>
054       *         9 = address of form .org .net .edu .gov ..biz -- official domains
055       */
056      private static int howValid(String email) { //NOPMD (reassign email)
057        if (email == null) {
058          return 0;
059        }
060        email = email.trim().toLowerCase();
061        int dotPlace = email.lastIndexOf('.');
062        if (0 < dotPlace && dotPlace < email.length() - 1) {
063          /* have at least x.y */
064          String tld = email.substring(dotPlace + 1);
065          if (badTLDs.contains(tld)) {
066            /* deliberate invalid address */
067            return 0;
068          }
069          // make sure none of fragments start or end in _ or -
070          String[] fragments = splitter.split(email);
071          boolean clean = true;
072          for (int i = 0; i < fragments.length; i++) {
073            if (fragments[i].startsWith("_") || fragments[i].endsWith("_")
074                || fragments[i].startsWith("-") || fragments[i].endsWith("-")) {
075              clean = false;
076              break;
077            }
078          } // end for
079          if (clean) {
080            Matcher m9 = p9.matcher(email);
081            if (m9.matches()) {
082              if (officialTLDs.contains(tld)) {
083                return 9;
084              }
085              else if (thisCountry.equals(tld)) {
086                return 8;
087              }
088              else if (nationalTLDs.contains(tld)) {
089                return 7;
090              }
091              else if (rareTLDs.contains(tld)) { //NOPMD
092                return 6;
093              }
094              else {
095                return 3; /* unknown tld */
096              }
097            }
098            // allow dots in name
099            Matcher m5 = p5.matcher(email);
100            if (m5.matches()) {
101              if (officialTLDs.contains(tld)) {
102                return 5;
103              }
104              else if (thisCountry.equals(tld)) {
105                return 5;
106              }
107              else if (nationalTLDs.contains(tld)) {
108                return 5;
109              }
110              else if (rareTLDs.contains(tld)) {
111                return 5;
112              }
113              else {
114                return 2; /* unknown tld */
115              }
116            }
117    
118            // IP
119            Matcher m4 = p4.matcher(email);
120            if (m4.matches()) {
121              return 4; /* can't tell TLD */ 
122            }
123    
124            // allow even lead/trail dots in name, except at start of domain
125            Matcher m3 = p3.matcher(email);
126            if (m3.matches()) {
127              if (officialTLDs.contains(tld)) {
128                return 3;
129              }
130              else if (thisCountry.equals(tld)) {
131                return 3;
132              }
133              else if (nationalTLDs.contains(tld)) {
134                return 3;
135              }
136              else if (rareTLDs.contains(tld)) {
137                return 3;
138              }
139              else {
140                return 2; /* unknown domain */ 
141              }
142            }
143          } // end if clean
144        }
145        // allow even unclean addresses, and addresses without a TLD to have a whack at passing RFC:822
146        try {
147    
148          /*
149           * see if InternetAddress likes it, it follows RFC:822. It will names without domains though.
150           */
151          InternetAddress.parse(email, true /* strict */);
152          // it liked it, no exception happened. Seems very sloppy.
153          return 1;
154        }
155        catch (AddressException e) {
156          // it did not like it
157          return 0;
158        }
159      }
160

162    
163      // allow _ - in name, lead and trailing ones are filtered later, no +.
164      private static Pattern p9 = Pattern.compile("[a-z0-9\\-_]++@[a-z0-9\\-_]++(\\.[a-z0-9\\-_]++)++");
165    
166      // to split into fields
167      private static Pattern splitter = Pattern.compile("[@\\.]");
168    
169      // to allow - _ dots in name, no +
170      private static Pattern p5 = Pattern
171          .compile("[a-z0-9\\-_]++(\\.[a-z0-9\\-_]++)*@[a-z0-9\\-_]++(\\.[a-z0-9\\-_]++)++");
172    
173      // IP style names, no +
174      private static Pattern p4 = Pattern
175          .compile("[a-z0-9\\-_]++(\\.[a-z0-9\\-_]++)*@\\[([0-9]{1,3}\\.){3}[0-9]{1,3}\\]");
176    
177      // allow dots anywhere, but not at start of domain name, no +
178      private static Pattern p3 = Pattern
179          .compile("[a-z0-9\\-_\\.]++@[a-z0-9\\-_]++(\\.[a-z0-9\\-_]++)++");
180    
181      /**
182       * build a HashSet from a array of String literals.
183       * 
184       * @param list array of strings
185       * @return HashSet you can use to test if a string is in the set.
186       */
187      private static HashSet<String> hmaker(String[] list) {
188        HashSet<String> map = new HashSet<String>(Math.max((int) (list.length / .75f) + 1, 16));
189        for (int i = 0; i < list.length; i++) {
190          map.add(list[i]);
191        }
192        return map;
193      }
