001    package org.hackystat.utilities.tstamp;
002    
003    import static org.junit.Assert.assertFalse;
004    import static org.junit.Assert.assertTrue;
005    
006    import javax.xml.datatype.XMLGregorianCalendar;
007    
008    import org.junit.Test;
009    
010    /**
011     * Tests the Tstamp class. 
012     * @author Philip Johnson
013     */
014    public class TestTstamp {
015      /**
016       * Tests the Tstamp class.
017       * @throws Exception if problems occur. 
018       */
019      @Test public void testTstampArithmetic () throws Exception {
020        XMLGregorianCalendar date1 = Tstamp.makeTimestamp("2007-08-01");
021        XMLGregorianCalendar date2 = Tstamp.makeTimestamp("2007-08-02");
022        XMLGregorianCalendar date3 = Tstamp.makeTimestamp("2007-08-03");
023        assertTrue("Test equal", Tstamp.equal(date1, date1));
024        assertTrue("Test increment days", Tstamp.equal(date2, Tstamp.incrementDays(date1, 1)));
025        
026        XMLGregorianCalendar date4 = Tstamp.makeTimestamp("2007-08-01T01:00:00");
027        assertTrue("Test increment hours", Tstamp.equal(date4, Tstamp.incrementHours(date1, 1)));
028        
029        XMLGregorianCalendar date5 = Tstamp.makeTimestamp("2007-08-01T00:01:00");
030        assertTrue("Test increment mins", Tstamp.equal(date5, Tstamp.incrementMinutes(date1, 1)));
031        
032        XMLGregorianCalendar date6 = Tstamp.makeTimestamp("2007-08-01T00:00:01");
033        assertTrue("Test increment secs", Tstamp.equal(date6, Tstamp.incrementSeconds(date1, 1)));
034    
035        assertTrue("Test greater than 1", Tstamp.greaterThan(date2, date1));
036        assertFalse("Test greater than 2", Tstamp.greaterThan(date1, date2));
037        assertFalse("Test greater than 3", Tstamp.greaterThan(date1, date1));
038        
039        assertTrue("Test less than 1", Tstamp.lessThan(date1, date2));
040        assertFalse("Test less than 2", Tstamp.lessThan(date2, date1));
041        assertFalse("Test less than 3", Tstamp.lessThan(date1, date1));   
042        
043        assertTrue("Test inbetween 1", Tstamp.inBetween(date1, date2, date3));
044        assertTrue("Test inbetween 2", Tstamp.inBetween(date1, date1, date2));
045        assertTrue("Test inbetween 3", Tstamp.inBetween(date1, date2, date2));    
046        assertFalse("Test inbetween 4", Tstamp.inBetween(date1, date3, date2));    
047      }
048    }
