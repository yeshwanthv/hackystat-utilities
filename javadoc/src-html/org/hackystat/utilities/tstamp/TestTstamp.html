001    package org.hackystat.utilities.tstamp;
002    
003    import static org.junit.Assert.assertEquals;
004    import static org.junit.Assert.assertFalse;
005    import static org.junit.Assert.assertTrue;
006    
007    import javax.xml.datatype.XMLGregorianCalendar;
008    
009    import org.junit.Test;
010    
011    /**
012     * Tests the Tstamp class. 
013     * @author Philip Johnson
014     */
015    public class TestTstamp {
016      /**
017       * Tests the Tstamp class.
018       * @throws Exception if problems occur. 
019       */
020      @Test public void testTstampArithmetic () throws Exception {
021        XMLGregorianCalendar date1 = Tstamp.makeTimestamp("2007-08-01");
022        XMLGregorianCalendar date2 = Tstamp.makeTimestamp("2007-08-02");
023        XMLGregorianCalendar date3 = Tstamp.makeTimestamp("2007-08-03");
024        assertTrue("Test equal", Tstamp.equal(date1, date1));
025        assertTrue("Test increment days", Tstamp.equal(date2, Tstamp.incrementDays(date1, 1)));
026        
027        XMLGregorianCalendar date4 = Tstamp.makeTimestamp("2007-08-01T01:00:00");
028        assertTrue("Test increment hours", Tstamp.equal(date4, Tstamp.incrementHours(date1, 1)));
029        
030        XMLGregorianCalendar date5 = Tstamp.makeTimestamp("2007-08-01T00:01:00");
031        assertTrue("Test increment mins", Tstamp.equal(date5, Tstamp.incrementMinutes(date1, 1)));
032        
033        XMLGregorianCalendar date6 = Tstamp.makeTimestamp("2007-08-01T00:00:01");
034        assertTrue("Test increment secs", Tstamp.equal(date6, Tstamp.incrementSeconds(date1, 1)));
035    
036        assertTrue("Test greater than 1", Tstamp.greaterThan(date2, date1));
037        assertFalse("Test greater than 2", Tstamp.greaterThan(date1, date2));
038        assertFalse("Test greater than 3", Tstamp.greaterThan(date1, date1));
039        
040        assertTrue("Test less than 1", Tstamp.lessThan(date1, date2));
041        assertFalse("Test less than 2", Tstamp.lessThan(date2, date1));
042        assertFalse("Test less than 3", Tstamp.lessThan(date1, date1));   
043        
044        assertTrue("Test inbetween 1", Tstamp.inBetween(date1, date2, date3));
045        assertTrue("Test inbetween 2", Tstamp.inBetween(date1, date1, date2));
046        assertTrue("Test inbetween 3", Tstamp.inBetween(date1, date2, date2));    
047        assertFalse("Test inbetween 4", Tstamp.inBetween(date1, date3, date2));   
048        
049        assertFalse("Test bogus time check", Tstamp.isBogusStartTime(date1));
050      }
051      
052      /**
053       * Tests time span. 
054       * 
055       * @throws Exception If problem occurs
056       */
057      @Test public void testTstampSpan() throws Exception {
058        XMLGregorianCalendar date1 = Tstamp.makeTimestamp("2007-08-01T01:00:00.000");
059        XMLGregorianCalendar date2 = Tstamp.makeTimestamp("2007-08-01T01:01:20.200");
060        XMLGregorianCalendar date3 = Tstamp.makeTimestamp("2007-08-01T01:01:20.400");
061        assertEquals("Test timespan 80200 ms", 80200, Tstamp.diff(date1, date2));
062        assertEquals("Test timespan 200 ms", 200, Tstamp.diff(date2, date3));
063      }
