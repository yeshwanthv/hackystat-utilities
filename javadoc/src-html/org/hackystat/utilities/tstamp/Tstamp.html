003    import java.util.GregorianCalendar;
004    import javax.xml.datatype.DatatypeConfigurationException;

009    /**
010     * Utility class that facilitates Timestamp representation and processing. There
011     * are too many classes already named "Timestamp", thus the abbreviated name.
012     * @author Philip Johnson
013     */
014    public class Tstamp {
015    
016      private static final String factoryErrorMsg = "Bad DataTypeFactory";
017    
018      /**
019       * Returns true if the passed string can be parsed into an
020       * XMLGregorianCalendar object.
021       * @param lexicalRepresentation The string representation.
022       * @return True if the string is a legal XMLGregorianCalendar.
023       */
024      public static boolean isTimestamp(String lexicalRepresentation) {
025        try {
026          DatatypeFactory factory = DatatypeFactory.newInstance();
027          factory.newXMLGregorianCalendar(lexicalRepresentation);
028          return true;
029    
030        }
031        catch (Exception e) {
032          return false;
033        }
034      }
035    
036      /**
037       * Returns an XMLGregorianCalendar, given its string representation.
038       * @param lexicalRepresentation The string representation.
039       * @return The timestamp.
040       * @throws Exception If the string cannot be parsed into a timestamp.
041       */
042      public static XMLGregorianCalendar makeTimestamp(String lexicalRepresentation)
043          throws Exception {
044        DatatypeFactory factory = DatatypeFactory.newInstance();
045        return factory.newXMLGregorianCalendar(lexicalRepresentation);
046      }
047    
048      /**
049       * Converts a javax.sql.Timestamp into a
050       * javax.xml.datatype.XMLGregorianCalendar.
051       * @param tstamp The javax.sql.Timestamp
052       * @return An new instance of a javax.xml.datatype.XmlGregorianCalendar
053       */
054      public static XMLGregorianCalendar makeTimestamp(java.sql.Timestamp tstamp) {
055        DatatypeFactory factory = null;
056        try {
057          factory = DatatypeFactory.newInstance();
058          GregorianCalendar calendar = new GregorianCalendar();
059          calendar.setTimeInMillis(tstamp.getTime());
060          return factory.newXMLGregorianCalendar(calendar);
061        }
062        catch (DatatypeConfigurationException e) {
063          throw new RuntimeException(factoryErrorMsg, e);
064        }
065      }
066    
067      /**
068       * Converts the specified time in milliseconds into a
069       * javax.xml.datatype.XMLGregorianCalendar.
070       * @param timeInMillis the specified time in milliseconds to convert.
071       * @return An new instance of a javax.xml.datatype.XmlGregorianCalendar
072       */
073      public static XMLGregorianCalendar makeTimestamp(long timeInMillis) {
074        DatatypeFactory factory = null;
075        try {
076          factory = DatatypeFactory.newInstance();

086      /**
087       * Returns a new XMLGregorianCalendar corresponding to the passed tstamp
088       * incremented by the number of days.
089       * @param tstamp The base date and time.
090       * @param days The number of days to increment. This can be a negative number.
091       * @return A new XMLGregorianCalendar instance representing the inc'd time.
092       */
093      public static XMLGregorianCalendar incrementDays(XMLGregorianCalendar tstamp, int days) {
094        DatatypeFactory factory = null;
095        try {
096          factory = DatatypeFactory.newInstance();
097          GregorianCalendar calendar = new GregorianCalendar();
098          long millis = tstamp.toGregorianCalendar().getTimeInMillis();
099          millis += 1000L * 60 * 60 * 24 * days;
100          calendar.setTimeInMillis(millis);
101          return factory.newXMLGregorianCalendar(calendar);
102        }
103        catch (DatatypeConfigurationException e) {
104          throw new RuntimeException(factoryErrorMsg, e);
105        }
106      }

241      /**
242       * Returns true if tstamp is equal to or between start and end.
243       * @param start The start time.
244       * @param tstamp The timestamp to test.
245       * @param end The end time.
246       * @return True if tstamp is between start and end.
247       */
248      public static boolean inBetween(XMLGregorianCalendar start, XMLGregorianCalendar tstamp,
249          XMLGregorianCalendar end) {
250        if ((Tstamp.equal(start, tstamp)) || (Tstamp.equal(end, tstamp))) {
251          return true;
252        }
253        if ((start.compare(tstamp) == DatatypeConstants.LESSER)
254            && (end.compare(tstamp) == DatatypeConstants.GREATER)) {
255          return true;
256        }
257        return false;
258      }

260      /**
261       * Returns true if time1 > time2.
262       * @param time1 The first time.
263       * @param time2 The second time.
264       * @return True if time1 > time2
265       */
266      public static boolean greaterThan(XMLGregorianCalendar time1, XMLGregorianCalendar time2) {
267        return time1.compare(time2) == DatatypeConstants.GREATER;
268      }
269    
270      /**
271       * Returns true if timeString1 > timeString2. Throws an unchecked
272       * IllegalArgument exception if the strings can't be converted to timestamps.
273       * @param timeString1 The first time.
274       * @param timeString2 The second time.
275       * @return True if time1 > time2
276       */
277      public static boolean greaterThan(String timeString1, String timeString2) {
278        try {
279          DatatypeFactory factory = DatatypeFactory.newInstance();
280          XMLGregorianCalendar time1 = factory.newXMLGregorianCalendar(timeString1);
281          XMLGregorianCalendar time2 = factory.newXMLGregorianCalendar(timeString2);
282          return time1.compare(time2) == DatatypeConstants.GREATER;
283        }
284        catch (Exception e) {
285          throw new IllegalArgumentException("Illegal timestring", e);
286        }
287      }
288    
289      /**
290       * Returns true if time1 < time2.
291       * @param time1 The first time.
292       * @param time2 The second time.
293       * @return True if time1 < time2
294       */
295      public static boolean lessThan(XMLGregorianCalendar time1, XMLGregorianCalendar time2) {
296        return time1.compare(time2) == DatatypeConstants.LESSER;
297      }
298    
299      /**
300       * Returns true if time1 equals time2. Note that this class compares for
301       * equality by converting to millis, which is apparently not what the built-in
302       * XMLGregorianCalendar class does, which leads to the XMLGregorianCalendar
303       * class saying that two logically equal instances are not equal. For example,
304       * 2007-01-01 does not equal 2007-01-01T00:00:00 using
305       * XMLGregorianCalendar.equals(). The Tstamp.equal() method does the right
306       * thing in this case.
307       * @param time1 The first time.
308       * @param time2 The second time.
309       * @return True if time1 equals time2
310       */
311      public static boolean equal(XMLGregorianCalendar time1, XMLGregorianCalendar time2) {
312        long millis1 = time1.toGregorianCalendar().getTimeInMillis();
313        long millis2 = time2.toGregorianCalendar().getTimeInMillis();
314        return millis1 == millis2;
315      }
