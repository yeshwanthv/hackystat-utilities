028       * Returns true if the passed string can be parsed into an
029       * XMLGregorianCalendar object.
030       * @param lexicalRepresentation The string representation.
031       * @return True if the string is a legal XMLGregorianCalendar.
032       */
033      public static boolean isTimestamp(String lexicalRepresentation) {
034        try {
035          DatatypeFactory factory = DatatypeFactory.newInstance();
036          factory.newXMLGregorianCalendar(lexicalRepresentation);
037          return true;
038    
039        }
040        catch (Exception e) {
041          return false;
042        }
043      }

045      /**
046       * Returns an XMLGregorianCalendar, given its string representation.
047       * Missing hours, minutes, second, millisecond, and timezone fields are given defaults.
048       * @param rep The string representation.
049       * @return The timestamp.
050       * @throws Exception If the string cannot be parsed into a timestamp.
051       */
052      public static XMLGregorianCalendar makeTimestamp(String rep)
053          throws Exception {
054        DatatypeFactory factory = DatatypeFactory.newInstance();
055        long mills = factory.newXMLGregorianCalendar(rep).toGregorianCalendar().getTimeInMillis();  
056        return makeTimestamp(mills);
057      }

059      /**
060       * Converts a javax.sql.Timestamp into a
061       * javax.xml.datatype.XMLGregorianCalendar.
062       * @param tstamp The javax.sql.Timestamp
063       * @return A new instance of a javax.xml.datatype.XmlGregorianCalendar
064       */
065      public static XMLGregorianCalendar makeTimestamp(java.sql.Timestamp tstamp) {
066        DatatypeFactory factory = null;
067        try {
068          factory = DatatypeFactory.newInstance();
069          GregorianCalendar calendar = new GregorianCalendar();
070          calendar.setTimeInMillis(tstamp.getTime());
071          return factory.newXMLGregorianCalendar(calendar);
072        }
073        catch (DatatypeConfigurationException e) {
074          throw new RuntimeException(factoryErrorMsg, e);
075        }
076      }

115      /**
116       * Returns a new XMLGregorianCalendar corresponding to the passed tstamp
117       * incremented by the number of days.
118       * @param tstamp The base date and time.
119       * @param days The number of days to increment. This can be a negative number.
120       * @return A new XMLGregorianCalendar instance representing the inc'd time.
121       */
122      public static XMLGregorianCalendar incrementDays(XMLGregorianCalendar tstamp, int days) {
123        DatatypeFactory factory = null;
124        try {
125          factory = DatatypeFactory.newInstance();
126          GregorianCalendar calendar = new GregorianCalendar();
127          long millis = tstamp.toGregorianCalendar().getTimeInMillis();
128          millis += 1000L * 60 * 60 * 24 * days;
129          calendar.setTimeInMillis(millis);
130          return factory.newXMLGregorianCalendar(calendar);
131        }
132        catch (DatatypeConfigurationException e) {
133          throw new RuntimeException(factoryErrorMsg, e);
134        }
135      }

294      /**
295       * In the early days of Hackystat, default project start times were 1000-01-01.
296       * This was stupid.  The following hack exists to correct projects containing this old 
297       * value.  This code and its callers can be removed when the disease is eradicated.
298       * @param startTime The startTime in question.
299       * @return True if it's before 1950.
300       */
301      public static boolean isBogusStartTime(XMLGregorianCalendar startTime) {
302        try {
303          XMLGregorianCalendar bogusTime = Tstamp.makeTimestamp("1950-01-01");
304          return Tstamp.lessThan(startTime, bogusTime);
305        }
306        catch (Exception e) {
307          return true;
308        }
309      }

311      /**
312       * Returns true if tstamp is equal to or between start and end.
313       * @param start The start time.
314       * @param tstamp The timestamp to test.
315       * @param end The end time.
316       * @return True if tstamp is between start and end.
317       */
318      public static boolean inBetween(XMLGregorianCalendar start, XMLGregorianCalendar tstamp,
319          XMLGregorianCalendar end) {
320        long startMillis = start.toGregorianCalendar().getTimeInMillis();
321        long endMillis = end.toGregorianCalendar().getTimeInMillis();
322        long tstampMillis = tstamp.toGregorianCalendar().getTimeInMillis();
323        return ((tstampMillis >= startMillis) && (tstampMillis <= endMillis));
324      }

338      /**
339       * Returns the number of days between time1 and time2.
340       * Returns a negative number if day1 is after day2.
341       * Takes into account daylight savings time issues. 
342       * @param day1 The first day.
343       * @param day2 The second day. 
344       * @return The number of days between the two days. 
345       */
346      public static int daysBetween(XMLGregorianCalendar day1, XMLGregorianCalendar day2) {
347        return (int) (getUnixDay(day2) - getUnixDay(day1));
348      }

350      /**
351       * Returns a long representing the number of days since the Unix epoch to the passed day.
352       * @param day The day of interest.
353       * @return The number of days since the epoch.
354       */
355      private static long getUnixDay(XMLGregorianCalendar day) {
356        GregorianCalendar greg = day.toGregorianCalendar();
357        long offset = greg.get(Calendar.ZONE_OFFSET) + greg.get(Calendar.DST_OFFSET);
358        long daysSinceEpoch = (long)Math.floor((double)(greg.getTime().getTime() + offset ) / 
359            ((double)MILLISECS_PER_DAY) );
360        return daysSinceEpoch;
361    }

368       * @param timeString2 The second time.

374          XMLGregorianCalendar time1 = factory.newXMLGregorianCalendar(timeString1);
375          XMLGregorianCalendar time2 = factory.newXMLGregorianCalendar(timeString2);

420      /**
421       * Returns true if the passed timestamp indicates some time today or some time 
422       * in the future. 
423       * 
424       * @param timestamp The timestamp of interest.
425       * @return True if it's today or some day in the future. 
426       */
427      public static boolean isTodayOrLater(XMLGregorianCalendar timestamp) {
428        XMLGregorianCalendar today = Tstamp.makeTimestamp();
429        boolean isToday = (today.getYear() == timestamp.getYear()) && 
430                          (today.getMonth() == timestamp.getMonth()) && 
431                          (today.getDay() == timestamp.getDay());
432        boolean afterToday = today.toGregorianCalendar().getTimeInMillis() <
433                           timestamp.toGregorianCalendar().getTimeInMillis();
434        return (isToday || afterToday);
435      }

437      /**
438       * Returns true if the passed timestamp indicates some time yesterday or some time 
439       * in the future. This is useful for the Commit/Churn DPDs, where the sensor typically
440       * sends data not from the current day, but from the day before. 
441       * 
442       * @param timestamp The timestamp of interest.
443       * @return True if it's today or some day in the future. 
444       */
445      public static boolean isYesterdayOrLater(XMLGregorianCalendar timestamp) {
446        XMLGregorianCalendar yesterday = Tstamp.incrementDays(Tstamp.makeTimestamp(), -1);
447    
448        boolean isYesterday = (yesterday.getYear() == timestamp.getYear()) && 
449                              (yesterday.getMonth() == timestamp.getMonth()) && 
450                              (yesterday.getDay() == timestamp.getDay());
451        boolean afterYesterday = yesterday.toGregorianCalendar().getTimeInMillis() <
452                                 timestamp.toGregorianCalendar().getTimeInMillis();
453        return (isYesterday || afterYesterday);
454      }

456      /**
457       * Returns a newly created sorted list of tstamps from the passed collection.
458       * @param tstamps The timestamps to be sorted. 
459       * @return A new list of tstamps, now in sorted order. 
460       */
461      public static List<XMLGregorianCalendar> sort(Collection<XMLGregorianCalendar> tstamps) {
462        List<XMLGregorianCalendar> tstampList = new ArrayList<XMLGregorianCalendar>(tstamps);
463        Collections.sort(tstampList, new TstampComparator());
464        return tstampList;
