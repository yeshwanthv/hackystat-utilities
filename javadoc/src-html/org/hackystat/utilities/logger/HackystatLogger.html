001    package org.hackystat.utilities.logger;
002    
003    import java.io.File;
004    import java.io.IOException;
005    import java.util.logging.ConsoleHandler;
006    import java.util.logging.FileHandler;
007    import java.util.logging.Level;
008    import java.util.logging.LogManager;
009    import java.util.logging.Logger;
010    
011    
012    /**
013     * Supports logging of informational and error messages by this service.
014     * @author Philip Johnson
015     */
016    public class HackystatLogger {
017      
018    
019      /**
020       * Create a new Logger for Hackystat.  
021       *
022       * @param loggerName The name of the logger to create.
023       */
024      private HackystatLogger(String loggerName) {
025        Logger logger = Logger.getLogger(loggerName);
026        logger.setUseParentHandlers(false);
027    
028        // Define a file handler that writes to the ~/.hackystat/logs directory, creating it if nec.
029        File logDir = new File(System.getProperty("user.home") + "/.hackystat/logs/");
030        logDir.mkdirs();
031        String fileName = logDir + "/" + loggerName + ".%u.log";
032        FileHandler fileHandler;
033        try {
034          fileHandler = new FileHandler(fileName, 500000, 1, true);
035          fileHandler.setFormatter(new OneLineFormatter());
036          logger.addHandler(fileHandler);
037        }
038        catch (IOException e) {
039          throw new RuntimeException("Could not open the log file for this Hackystat service.", e);
040        }
041    
042        // Define a console handler to also write the message to the console.
043        ConsoleHandler consoleHandler = new ConsoleHandler();
044        consoleHandler.setFormatter(new OneLineFormatter());
045        logger.addHandler(consoleHandler);
046        setLoggingLevel(logger, "INFO");
047      }
048    
049      
050      /**
051       * Return the Hackystat Logger named with loggerName, creating it if it does not yet exist.
052       * Hackystat loggers have the following characteristics:
053       * <ul>
054       * <li> Log messages are one line and are prefixed with a time stamp using the OneLineFormatter
055       * class.
056       * <li> The logger creates a Console logger and a File logger. 
057       * <li> The File logger is written out to the ~/.hackystat/logs/ directory, creating this if
058       * it is not found.
059       * <li> The File log name is {name}.%u.log.
060       * </ul> 
061       * @param loggerName The name of this HackystatLogger.
062       * @return The Logger instance. 
063       */
064      public static Logger getLogger(String loggerName) {
065        Logger logger = LogManager.getLogManager().getLogger(loggerName);
066        if (logger == null) {
067          new HackystatLogger(loggerName);
068        }
069        return LogManager.getLogManager().getLogger(loggerName);
070      }
071    
072      /**
073       * Sets the logging level to be used for this Hackystat logger.
074       * If the passed string cannot be parsed into a Level, then INFO is set by default.
075       * @param logger The logger whose level is to be set.  
076       * @param level The new Level.
077       */
078      public static void setLoggingLevel(Logger logger, String level) {
079        Level newLevel = Level.INFO;
080        try {
081          newLevel = Level.parse(level);
082        }
083        catch (Exception e) {
084          logger.info("Couldn't set Logging level to: " + level);
085        }
086        logger.setLevel(newLevel);
087        logger.getHandlers()[0].setLevel(newLevel);
088        logger.getHandlers()[1].setLevel(newLevel);
089      }
090    }
