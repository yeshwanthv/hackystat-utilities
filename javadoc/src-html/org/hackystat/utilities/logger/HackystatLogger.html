001    package org.hackystat.utilities.logger;
002    
003    import java.io.File;
004    import java.io.IOException;
005    import java.util.logging.ConsoleHandler;
006    import java.util.logging.FileHandler;
007    import java.util.logging.Level;
008    import java.util.logging.LogManager;
009    import java.util.logging.Logger;
010    
011    import org.hackystat.utilities.home.HackystatUserHome;
012    
013    
014    /**
015     * Supports logging of informational and error messages by a service.
016     * Uses HackystatUserHome.getHome() to determine where to put the logs directory.
017     * @author Philip Johnson
018     */
019    public class HackystatLogger {
020      
021    
022      /**
023       * Create a new Logger for Hackystat.  
024       *
025       * @param loggerName The name of the logger to create.
026       * @param subdir If non-null, then logging files are placed in .hackystat/[subdir]/logs.
027       * Otherwise they are placed in .hackystat/logs.
028       */
029      private HackystatLogger(String loggerName, String subdir) {
030        Logger logger = Logger.getLogger(loggerName);
031        logger.setUseParentHandlers(false);
032    
033        // Define a file handler that writes to the ~/.hackystat/logs directory, creating it if nec.
034        String logSubDir = (subdir == null) ? ".hackystat/logs/" : ".hackystat/" + subdir + "/logs";
035        File logDir = new File(HackystatUserHome.getHome() + logSubDir);
036        logDir.mkdirs();
037        String fileName = logDir + "/" + loggerName + ".%u.log";
038        FileHandler fileHandler;
039        try {
040          fileHandler = new FileHandler(fileName, 500000, 1, true);
041          fileHandler.setFormatter(new OneLineFormatter());
042          logger.addHandler(fileHandler);
043        }
044        catch (IOException e) {
045          throw new RuntimeException("Could not open the log file for this Hackystat service.", e);
046        }
047    
048        // Define a console handler to also write the message to the console.
049        ConsoleHandler consoleHandler = new ConsoleHandler();
050        consoleHandler.setFormatter(new OneLineFormatter());
051        logger.addHandler(consoleHandler);
052        setLoggingLevel(logger, "INFO");
053      }
054    
055      
056      /**
057       * Return the Hackystat Logger named with loggerName, creating it if it does not yet exist.
058       * Hackystat loggers have the following characteristics:
059       * <ul>
060       * <li> Log messages are one line and are prefixed with a time stamp using the OneLineFormatter
061       * class.
062       * <li> The logger creates a Console logger and a File logger. 
063       * <li> The File logger is written out to the ~/.hackystat/logs/ directory, creating this if
064       * it is not found.
065       * <li> The File log name is {name}.%u.log.
066       * </ul> 
067       * @param loggerName The name of this HackystatLogger.
068       * @return The Logger instance. 
069       */
070      public static Logger getLogger(String loggerName) {
071        return HackystatLogger.getLogger(loggerName, null);
072      }
073      
074      /**
075       * Return the Hackystat Logger named with loggerName, creating it if it does not yet exist.
076       * Hackystat loggers have the following characteristics:
077       * <ul>
078       * <li> Log messages are one line and are prefixed with a time stamp using the OneLineFormatter
079       * class.
080       * <li> The logger creates a Console logger and a File logger. 
081       * <li> The File logger is written out to the ~/.hackystat/[subDir]/logs/ directory, creating 
082       * this if it is not found.
083       * <li> The File log name is {name}.%u.log.
084       * </ul> 
085       * @param loggerName The name of this HackystatLogger.
086       * @param subDir The .hackystat subdirectory in which the log/ directory should be put.
087       * @return The Logger instance. 
088       */
089      public static Logger getLogger(String loggerName, String subDir) {
090        Logger logger = LogManager.getLogManager().getLogger(loggerName);
091        if (logger == null) {
092          new HackystatLogger(loggerName, subDir);
093        }
094        return LogManager.getLogManager().getLogger(loggerName);
095      }
096    
097      /**
098       * Sets the logging level to be used for this Hackystat logger.
099       * If the passed string cannot be parsed into a Level, then INFO is set by default.
100       * @param logger The logger whose level is to be set.  
101       * @param level The new Level.
102       */
103      public static void setLoggingLevel(Logger logger, String level) {
104        Level newLevel = Level.INFO;
105        try {
106          newLevel = Level.parse(level);
107        }
108        catch (Exception e) {
109          logger.info("Couldn't set Logging level to: " + level);
110        }
111        logger.setLevel(newLevel);
112        logger.getHandlers()[0].setLevel(newLevel);
113        logger.getHandlers()[1].setLevel(newLevel);
114      }
115    }
