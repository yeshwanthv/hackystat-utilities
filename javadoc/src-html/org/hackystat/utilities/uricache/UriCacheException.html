001    package org.hackystat.utilities.uricache;
002    
003    /**
004     * A cache operations exception.
005     * 
006     * @author <a href="mailto:seninp@gmail.com">Pavel Senin<a>
007     */
008    public class UriCacheException extends Exception {
009    
010      /**
011       * Fancy eclipse serial.
012       */
013      private static final long serialVersionUID = 1L;
014    
015      /**
016       * An empty implementation.
017       */
018      public UriCacheException() {
019        assert true;
020      }
021    
022      /**
023       * Messaged implementation.
024       * 
025       * @param msg the message.
026       */
027      public UriCacheException(String msg) {
028        super(msg);
029      }
030    
031      /**
032       * The basic exception translator.
033       * 
034       * @param cause throwable which causes this exception.
035       */
036      public UriCacheException(Throwable cause) {
037        super(cause);
038      }
039    
040      /**
041       * The full format exception.
042       * 
043       * @param msg the exception message.
044       * @param cause throwable which causes this exception.
045       */
046      public UriCacheException(String msg, Throwable cause) {
047        super(msg, cause);
048      }
049    
050    }
