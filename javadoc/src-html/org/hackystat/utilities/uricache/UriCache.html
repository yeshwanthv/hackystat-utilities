019    /**
020     * Provides a wrapper around Apache JCS (Java Caching System) to facilitate Hackystat caching. This
021     * wrapper provides the following:
022     * <ul>
023     * <li> Automatic configuration of an indexed disk cache backing store.
024     * <li> Write-through caching: All cached instances are written out to disk. 
025     * <li> Provides a default maximum life for expiring of entries of one day.
026     * <li> Provides a default maximum cache size of 10000 instances.
027     * <li> Provides a default directory location (inside ~/.hackystat) for backing store files. 
028     * <li> Helps ensure that all UriCache instances have a unique name.
029     * <li> All caches use the JCS "group" facility to allow access to the set of keys. 
030     * <li> Constructor uses "days" rather than seconds as time unit for maxLife.
031     * <li> put() uses "hours" rather than seconds as time unit for maxLife.
032     * <li> A more convenient API for setting/getting items from the cache and controlling logging.
033     * <li> Logging of exceptions raised by JCS.
034     * <li> Disables JCS logging messages unless the System property
035     * org.hackystat.utilities.uricache.enableJCSLogging is set.
036     * <li> Shutdown hook ensures that backing index file is closed correctly on JVM exit. 
037     * <li> Convenient packaging mechanism for required jar files to simplify library use.
038     * </ul>
039     * 
040     * Here's an example usage, where we create a separate cache for each user to hold their sensor data
041     * instances as part of the dailyprojectdata service.
042     * 
043     * <pre>
044     * SensorBaseClient client = new SensorBaseClient(user, host);
045     * UriCache cache = new UriCache(user.getEmail(), &quot;dailyprojectdata&quot;);
046     *   :
047     * SensorData data = (SensorData)cache.get(uriString);
048     * if (data == null) {
049     *   // Cache doesn't have it, so retrieve from SensorBase and cache locally for next time.
050     *   data = client.getSensorData(uriString);
051     *   cache.put(uriString, data);
052     * }
053     * </pre>
054     * 
055     * The cache files are in the directory ~/.hackystat/dailyprojectdata/uricache. Instances expire
056     * from the cache after one day, by default. The maximum number of in-memory instances is 10,000, by
057     * default.
058     * 
059     * @author Philip Johnson
060     */
061    public class UriCache {
062      
063      /** The number of seconds in a day. * */
064      private static final long secondsInADay = 60L * 60L * 24L;
065      /** Maximum life in seconds that an entry stays in the cache. Default is 1 day. */
066      private static final Long defaultMaxLifeSeconds = secondsInADay;
067      /** Maximum number of in-memory instances before sending items to disk. Default is 50,000. */
068      private static final Long defaultCapacity = 10000L;
069      /** The name of this cache, which defines a "region" in JCS terms. */
070      private String cacheName = null;
071      /** The logger used for cache exception logging. */
072      private Logger logger = null;
073      /** Holds a list of already defined caches to help ensure uniqueness. */
074      private static ArrayList<String> cacheNames = new ArrayList<String>();
075      /** Default group name. No client should ever using the following string for a group. */
076      private static final String DEFAULT_GROUP = "__Default_UriCache_Group__";
077      
078      /** A thread that will ensure that all of these caches will be disposed of during shutdown. */ 
079      private static Thread shutdownThread = new Thread() {
080        /** Run the shutdown hook for disposing of all caches. */
081        @Override 
082        public void run() {
083          for (String cacheName : cacheNames) {
084            try {
085              System.out.println("Shutting down " + cacheName + " cache.");
086              JCS.getInstance(cacheName).dispose();
087            }
088            catch (Exception e) {
089              String msg = "Failure to clear cache " + cacheName + ":" + e.getMessage();
090              System.out.println(msg);
091            }
092          }
093        }
094      };
095      
096      /** A boolean that enables us to figure out whether to install the shutdown thread.  */
097      private static boolean hasShutdownHook = false;  //NOPMD
098      
099      /**
100       * Creates a new UriCache instance with the specified name. Good for services who want to create a
101       * single cache for themselves, such as "dailyprojectdata". If the cache with this name has been 
102       * previously created, then this instance will become an alias to the previously existing cache. 
103       * 
104       * @param cacheName The name of this cache.
105       * @param subDir the .hackystat subdirectory in which the uricache directory holding the backing
106       *        store will be created.
107       */
108      public UriCache(String cacheName, String subDir) {
109        this(cacheName, subDir, new Double(defaultMaxLifeSeconds), defaultCapacity);
110      }
111      
112      /**
113       * Creates a new UriCache with the specified parameters. 
114       * If a cache with this name already exists, then this instance will be an alias to that cache
115       * and its original configuration will remain unchanged. 
116       * 
117       * @param cacheName The name of this UriCache, which will be used as the JCS "region" and also
118       *        define the subdirectory in which the index files will live.
119       * @param subDir the .hackystat subdirectory in which the uricache directory holding the backing
120       *        store will be created.
121       * @param maxLifeDays The maximum number of days after which items expire from the cache.
122       * @param capacity The maximum number of instances to hold in the cache. 
123       */
124      public UriCache(String cacheName, String subDir, Double maxLifeDays, Long capacity) {
125        // Set up the shutdown hook if we're the first one. Not thread safe, but there's not too
126        // much harm done if there are multiple shutdown hooks running.
127        if (!UriCache.hasShutdownHook) {
128          Runtime.getRuntime().addShutdownHook(UriCache.shutdownThread);
129          UriCache.hasShutdownHook = true;
130        }
131        this.cacheName = cacheName;
132        this.logger = HackystatLogger.getLogger(cacheName + ".uricache", subDir);
133    
134        // Finish configuration if this is a new instance of the cache.
135        if (!UriCache.cacheNames.contains(cacheName)) {
136          UriCache.cacheNames.add(cacheName);
137          if (!System.getProperties().containsKey(
138              "org.hackystat.utilities.uricache.enableJCSLogging")) {
139            Logger.getLogger("org.apache.jcs").setLevel(Level.OFF);
140          }
141          CompositeCacheManager ccm = CompositeCacheManager.getUnconfiguredInstance();
142          long maxLifeSeconds = (long) (maxLifeDays * secondsInADay);
143          ccm.configure(initJcsProps(cacheName, subDir, maxLifeSeconds, capacity));
144        }
145      }
146      
147      /**
148       * Adds the key-value pair to this cache. Entry will expire from cache after the default maxLife
149       * (currently 24 hours). Logs a message if the cache throws an exception.
150       * 
151       * @param key The key, typically a UriString.
152       * @param value The value, typically the object returned from the Hackystat service.
153       */
154      public void put(Serializable key, Serializable value) {
155        try {
156          JCS.getInstance(this.cacheName).putInGroup(key, DEFAULT_GROUP, value);
157        }
158        catch (CacheException e) {
159          String msg = "Failure to add " + key + " to cache " + this.cacheName + ":" + e.getMessage();
160          this.logger.warning(msg);
161        }
162      }
163      
164      /**
165       * Adds the key-value pair to this cache with an explicit expiration time.  
166       * 
167       * @param key The key, typically a UriString.
168       * @param value The value, typically the object returned from the Hackystat service.
169       * @param maxLifeHours The number of hours before this item will expire from cache.
170       */
171      public void put(Serializable key, Serializable value, double maxLifeHours) {
172        try {
173          ElementAttributes attributes = new ElementAttributes();
174          long maxLifeSeconds = (long)(maxLifeHours * 3600D);
175          attributes.setMaxLifeSeconds(maxLifeSeconds);
176          attributes.setIsEternal(false);
177          JCS.getInstance(this.cacheName).putInGroup(key, DEFAULT_GROUP, value, attributes);
178        }
179        catch (CacheException e) {
180          String msg = "Failure to add " + key + " to cache " + this.cacheName + ":" + e.getMessage();
181          this.logger.warning(msg);
182        }
183      }
184      
185      /**
186       * Returns the object associated with key from the cache, or null if not found. 
187       * 
188       * @param key The key whose associated value is to be retrieved.
189       * @return The value, or null if not found.
190       */
191      public Object get(Serializable key) {
192        try {
193          return JCS.getInstance(this.cacheName).getFromGroup(key, DEFAULT_GROUP);
194        }
195        catch (CacheException e) {
196          String msg = "Failure of get: " + key + " in cache " + this.cacheName + ":" + e.getMessage();
197          this.logger.warning(msg);
198          return null;
199        }
200      }
201    
202      /**
203       * Ensures that the key-value pair associated with key is no longer in this cache. 
204       * Logs a message if the cache throws an exception.
205       * 
206       * @param key The key to be removed.
207       */
208      public void remove(Serializable key) {
209        try {
210          JCS.getInstance(this.cacheName).remove(key, DEFAULT_GROUP);
211        }
212        catch (CacheException e) {
213          String msg = "Failure to remove: " + key + " cache " + this.cacheName + ":" + e.getMessage();
214          this.logger.warning(msg);
215        }
216      }
217      
218      /**
219       * Removes everything in the default cache, but not any of the group caches. 
220       */
221      public void clear() {
222        clearGroup(DEFAULT_GROUP);
223      }
224    
225      /**
226       * Clears the default as well as all group caches. 
227       */
228      public void clearAll() {
229        try {
230          JCS.getInstance(this.cacheName).clear();
231        }
232        catch (CacheException e) {
233          String msg = "Failure to clear cache " + this.cacheName + ":" + e.getMessage();
234          this.logger.warning(msg);
235        }
236      }

238      /**
239       * Returns the set of keys associated with this cache. 
240       * @return The set containing the keys for this cache. 
241       */
242      public Set<Serializable> getKeys() {
243        return getGroupKeys(DEFAULT_GROUP);
244      }
245      
246      /**
247       * Returns the current number of elements in this cache. 
248       * @return The current size of this cache. 
249       */
250      public int size() {
251        return getGroupSize(DEFAULT_GROUP);
252      }

255      /**
256       * Shuts down the specified cache, and removes it from the list of active caches so it can be
257       * created again.
258       * 
259       * @param cacheName The name of the cache to dispose of.
260       */
261      public static void dispose(String cacheName) {
262        try {
263          cacheNames.remove(cacheName);
264          JCS.getInstance(cacheName).dispose();
265        }
266        catch (CacheException e) {
267          String msg = "Failure to clear cache " + cacheName + ":" + e.getMessage();
268          System.out.println(msg);
269        }
270      }

273      /**
274       * Implements group-based addition of cache elements.
275       * @param key The key.
276       * @param group The group.
277       * @param value The value.
278       */
279      public void putInGroup(Serializable key, String group, Serializable value) {
280        try {
281          JCS.getInstance(this.cacheName).putInGroup(key, group, value);
282        }
283        catch (CacheException e) {
284          String msg = "Failure to add " + key + " to cache " + this.cacheName + ":" + e.getMessage();
285          this.logger.warning(msg);
286        }
287      }

289      /**
290       * Implements group-based retrieval of cache elements. 
291       * @param key The key.
292       * @param group The group.
293       * @return The element associated with key in the group, or null.
294       */
295      public Object getFromGroup(Serializable key, String group) {
296        try {
297          return JCS.getInstance(this.cacheName).getFromGroup(key, group);
298        }
299        catch (CacheException e) {
300          String msg = "Failure of get: " + key + " in cache " + this.cacheName + ":" + e.getMessage();
301          this.logger.warning(msg);
302          return null;
303        }
304      }
305      
306      /**
307       * Implements group-based removal of cache elements. 
308       * @param key The key whose value is to be removed. 
309       * @param group The group.
310       */
311      public void removeFromGroup(Serializable key, String group) {
312        try {
313          JCS.getInstance(this.cacheName).remove(key, group);
314        }
315        catch (CacheException e) {
316          String msg = "Failure to remove: " + key + " cache " + this.cacheName + ":" + e.getMessage();
317          this.logger.warning(msg);
318        }
319      }
320      
321      /**
322       * Returns the set of cache keys associated with this group.
323       * @param group The group.
324       * @return The set of cache keys for this group.
325       */
326      @SuppressWarnings("unchecked")
327      public Set<Serializable> getGroupKeys(String group) {
328        Set<Serializable> keySet;
329        try {
330          keySet = JCS.getInstance(this.cacheName).getGroupKeys(group);
331        }
332        catch (CacheException e) {
333          String msg = "Failure to obtain keyset for cache: " + this.cacheName;
334          this.logger.warning(msg);
335          keySet = new HashSet<Serializable>();
336        }
337        return keySet;
338      }
339      
340      /**
341       * Returns the current number of elements in this cache group.
342       * @param group The name of the group.  
343       * @return The current size of this cache. 
344       */
345      public int getGroupSize(String group) {
346        return getGroupKeys(group).size();
347      }
348     
349      /**
350       * Removes everything in the default cache, but not any of the group caches.
351       * @param group The group name.  
352       */
353      public void clearGroup(String group) {
354        try {
355          JCS cache = JCS.getInstance(this.cacheName);
356          for (Object key : cache.getGroupKeys(group)) {
357            cache.remove(key, group);
358          }
359        }
360        catch (CacheException e) {
361          String msg = "Failure to clear cache " + this.cacheName + ":" + e.getMessage();
362          this.logger.warning(msg);
363        }
364      }

366      /**
367       * Sets up the Properties instance for configuring this JCS cache instance. Each UriCache is
368       * defined as a JCS "region". Given a UriCache named "PJ", we create a properties instance whose
369       * contents are similar to the following:
370       * 
371       * <pre>
372       * jcs.region.PJ=DC-PJ
373       * jcs.region.PJ.cacheattributes=org.apache.jcs.engine.CompositeCacheAttributes
374       * jcs.region.PJ.cacheattributes.MaxObjects=[maxCacheCapacity]
375       * jcs.region.PJ.cacheattributes.MemoryCacheName=org.apache.jcs.engine.memory.lru.LRUMemoryCache
376       * jcs.region.PJ.cacheattributes.UseMemoryShrinker=true
377       * jcs.region.PJ.cacheattributes.MaxMemoryIdleTimeSeconds=3600
378       * jcs.region.PJ.cacheattributes.ShrinkerIntervalSeconds=3600
379       * jcs.region.PJ.cacheattributes.MaxSpoolPerRun=500
380       * jcs.region.PJ.elementattributes=org.apache.jcs.engine.ElementAttributes
381       * jcs.region.PJ.elementattributes.IsEternal=false
382       * jcs.region.PJ.elementattributes.MaxLifeSeconds=[maxIdleTime]
383       * jcs.auxiliary.DC-PJ=org.apache.jcs.auxiliary.disk.indexed.IndexedDiskCacheFactory
384       * jcs.auxiliary.DC-PJ.attributes=org.apache.jcs.auxiliary.disk.indexed.IndexedDiskCacheAttributes
385       * jcs.auxiliary.DC-PJ.attributes.DiskPath=[cachePath]
386       * jcs.auxiliary.DC-PJ.attributes.maxKeySize=10000000
387       * </pre>
388       * 
389       * We define cachePath as HackystatHome.getHome()/.hackystat/[cacheSubDir]/cache. This enables a
390       * service a cache name of "dailyprojectdata" and have the cache data put inside its internal
391       * subdirectory.
392       * 
393       * See bottom of: http://jakarta.apache.org/jcs/BasicJCSConfiguration.html for more details.
394       * 
395       * @param cacheName The name of this cache, used to define the region properties.
396       * @param subDir The subdirectory name, used to generate the disk storage directory.
397       * @param maxLifeSeconds The maximum life of instances in the cache in seconds before they expire.
398       * @param maxCapacity The maximum size of this cache.
399       * @return The properties file.
400       */
401      private Properties initJcsProps(String cacheName, String subDir, Long maxLifeSeconds, 
402          Long maxCapacity) {
403        String reg = "jcs.region." + cacheName;
404        String regCacheAtt = reg + ".cacheattributes";
405        String regEleAtt = reg + ".elementattributes";
406        String aux = "jcs.auxiliary.DC-" + cacheName;
407        String auxAtt = aux + ".attributes";
408        String memName = "org.apache.jcs.engine.memory.lru.LRUMemoryCache";
409        String diskAttName = "org.apache.jcs.auxiliary.disk.indexed.IndexedDiskCacheAttributes";
410        Properties props = new Properties();
411        props.setProperty(reg, "DC-" + cacheName);
412        props.setProperty(regCacheAtt, "org.apache.jcs.engine.CompositeCacheAttributes");
413        props.setProperty(regCacheAtt + ".MaxObjects", maxCapacity.toString());
414        props.setProperty(regCacheAtt + ".MemoryCacheName", memName);
415        props.setProperty(regCacheAtt + ".UseMemoryShrinker", "true");
416        props.setProperty(regCacheAtt + ".MaxMemoryIdleTimeSeconds", "3600");
417        props.setProperty(regCacheAtt + ".ShrinkerIntervalSeconds", "3600");
418        props.setProperty(regCacheAtt + ".DiskUsagePatternName", "UPDATE");
419        props.setProperty(regCacheAtt + ".MaxSpoolPerRun", "500");
420        props.setProperty(regEleAtt, "org.apache.jcs.engine.ElementAttributes");
421        props.setProperty(regEleAtt + ".IsEternal", "false");
422        props.setProperty(regEleAtt + ".MaxLifeSeconds", maxLifeSeconds.toString());
423        props.setProperty(aux, "org.apache.jcs.auxiliary.disk.indexed.IndexedDiskCacheFactory");
424        props.setProperty(auxAtt, diskAttName);
425        props.setProperty(auxAtt + ".DiskPath", getCachePath(subDir));
426        props.setProperty(auxAtt + ".maxKeySize", "1000000");
427        return props;
428      }

430      /**
431       * Returns the fully qualified file path to the directory in which the backing store files for
432       * this cache will be placed. Creates the path if it does not already exist.
433       * 
434       * @param cacheSubDir The subdirectory where we want to locate the cache files.
435       * @return The fully qualified file path to the location where we should put the index files.
436       */
437      private String getCachePath(String cacheSubDir) {
438        File path = new File(HackystatUserHome.getHome(), ".hackystat/" + cacheSubDir + "/uricache");
439        path.mkdirs();
440        return path.getAbsolutePath();
441      }
