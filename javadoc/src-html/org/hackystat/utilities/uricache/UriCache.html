001    package org.hackystat.utilities.uricache;
002    
003    import java.io.Serializable;
004    import java.util.ArrayList;
005    import java.util.HashMap;
006    import java.util.Map;
007    import java.util.Properties;
008    import java.util.logging.Level;
009    import java.util.logging.Logger;
010    
011    import javax.xml.datatype.XMLGregorianCalendar;
012    
013    import org.apache.jcs.auxiliary.disk.indexed.IndexedDiskCache;
014    import org.apache.jcs.auxiliary.disk.indexed.IndexedDiskCacheAttributes;
015    import org.apache.jcs.engine.CacheElement;
016    import org.apache.jcs.engine.ElementAttributes;
017    import org.apache.jcs.engine.behavior.ICacheElement;
018    import org.apache.jcs.engine.behavior.IElementAttributes;
019    import org.apache.jcs.engine.stats.behavior.IStatElement;
020    import org.apache.jcs.engine.stats.behavior.IStats;
021    
022    /**
023     * Provides a caching mechanism based upon Apache JCS (Java Caching System). 
024     * 
025     * @author <a href="mailto:seninp@gmail.com">Pavel Senin<a>
026     */
027    public class UriCache {
028    
029      /** JCS cache handler. */
030      private IndexedDiskCache uriCache;
031    
032      /** JCS cache name. */
033      private String cacheName;
034    
035      /** JCS cache properties set. */
036      private Properties cacheProperties;
037    
038      /** JCS element "life time". */
039      private Long maxIdleTime;
040    
041      /** JCS instances name holder */
042      private static ArrayList<String> cacheNames = new ArrayList<String>();
043    
044      /**
045       * Constructor, an instance of UriCache configured according to the Properties provided.
046       * 
047       * @param cacheName the name used for this cache identification and to name the 
048       * file containing this cache.
049       * @param cacheProperties the cache configuration properties.
050       * @throws UriCacheException If the cacheName is already in use.
051       */
052      public UriCache(String cacheName, UriCacheProperties cacheProperties) throws UriCacheException {
053    
054        // getting rid of DEBUG level log messages
055        Level loggerLevel = cacheProperties.getLoggerLevel();
056        Logger logger = Logger.getLogger("org.apache.jcs");
057        logger.setLevel(loggerLevel);
058    
059        this.cacheName = cacheName;
060    
061        // check for the name duplication in here
062        if (UriCache.cacheNames.contains(this.cacheName)) {
063          throw new UriCacheException(
064              "Error while attemting get a cache instance: the cache region name " + cacheName
065                  + " is in use.");
066        }
067        else {
068          UriCache.cacheNames.add(this.cacheName);
069        }
070    
071        this.cacheProperties = cacheProperties.getProperties();
072        this.maxIdleTime = Long.valueOf(this.cacheProperties
073            .getProperty("jcs.region.UriCache.elementattributes.MaxLifeSeconds"));
074    
075        IndexedDiskCacheAttributes cattr = new IndexedDiskCacheAttributes();
076        cattr.setCacheName(this.cacheName);
077        cattr.setMaxKeySize(Integer.valueOf(this.cacheProperties
078            .getProperty("jcs.region.UriCache.cacheattributes.MaxObjects")));
079        cattr.setDiskPath(cacheProperties.getCacheStoragePath());
080        this.uriCache = new IndexedDiskCache(cattr);
081      }
082    
083      /**
084       * Place a new item in the cache, associated with uriString. If there is currently an item
085       * associated with the same uriString it is replaced.
086       * 
087       * @param uriString identity (URI) of the object to cache.
088       * @param obj The object to cache.
089       * @throws UriCacheException in case of error.
090       */
091      public void cache(Serializable uriString, Serializable obj) throws UriCacheException {
092        //
093        // I'm hardcoding parameters here, probably good move is to get these from properties when
094        // everything will be working.
095        //
096        IElementAttributes eAttr = new ElementAttributes();
097        eAttr.setIdleTime(this.maxIdleTime);
098        eAttr.setIsSpool(true);
099        eAttr.setIsEternal(false);
100        eAttr.setIsLateral(false);
101        eAttr.setIsRemote(false);
102        eAttr.setIsSpool(true);
103        ICacheElement element = new CacheElement(this.cacheName, uriString, obj);
104        element.setElementAttributes(eAttr);
105        this.uriCache.doUpdate(element);
106      }
107    
108      /**
109       * Puts object into cache and specifies object lifetime within the cache.
110       * 
111       * @param uriString Identity of the object to cache.
112       * @param obj Object to cache.
113       * @param expirationTimeStamp Object expiration timestamp, object will be removed from the cache
114       *        once System time is greater than the timestamp value.
115       * @throws UriCacheException in case of error.
116       */
117      public void cache(Serializable uriString, Serializable obj,
118          XMLGregorianCalendar expirationTimeStamp) throws UriCacheException {
119    
120        // first of all calculating the life time of this object from now
121        Long currTime = System.currentTimeMillis();
122        Long lifeTime = ((Integer) expirationTimeStamp.getMillisecond()).longValue() - currTime;
123    
124        IElementAttributes eAttr = new ElementAttributes();
125        eAttr.setIdleTime(lifeTime);
126        eAttr.setIsSpool(true);
127        eAttr.setIsEternal(false);
128        eAttr.setIsLateral(false);
129        eAttr.setIsRemote(false);
130        eAttr.setIsSpool(true);
131        ICacheElement element = new CacheElement(this.cacheName, uriString, obj);
132        element.setElementAttributes(eAttr);
133        this.uriCache.doUpdate(element);
134      }
135    
136      /**
137       * Removes all elements from the cache.
138       * 
139       * @throws UriCacheException in case of error.
140       */
141      public void clear() throws UriCacheException {
142        this.uriCache.doRemoveAll();
143      }
144    
145      /**
146       * ShutDowns the cache, flushes all items from the memory to cache and clears memory region.
147       * 
148       */
149      public void shutdown() {
150        this.uriCache.dispose();
151        UriCache.cacheNames.remove(this.cacheName);
152      }
153    
154      /**
155       * Retrieves object from the cache.
156       * 
157       * @param uriString URI of the object to search for.
158       * @return The cached object or <em>null</em> if not found.
159       */
160      public Serializable lookup(Serializable uriString) {
161        ICacheElement ce = this.uriCache.get(uriString);
162        if (null == ce) {
163          return null;
164        }
165        else {
166          return ce.getVal();
167        }
168      }
169    
170      /**
171       * Removes an Object from the cache.
172       * 
173       * @param uriString Identity of the object to be removed.
174       */
175      public void remove(Serializable uriString)  {
176        this.uriCache.remove(uriString);
177      }
178    
179      /**
180       * Sets the cache auto-expire time. Cache will auto expire elements after specified seconds to
181       * reclaim space.
182       * 
183       * @param seconds The new ShrinkerIntervalSeconds value.
184       */
185      public void setMaxMemoryIdleTimeSeconds(Long seconds) {
186        this.maxIdleTime = seconds;
187      }
188    
189      /**
190       * Reports the cache memory region name.
191       * 
192       * @return cache name.
193       */
194      public String getName() {
195        return this.cacheName;
196      }
197    
198      /**
199       * Reports cache statistics:
200       * <ul>
201       * <li>Cache type</li>
202       * <li>Is Alive</li>
203       * <li>Key Map Size</li>
204       * <li>Data File Length</li>
205       * <li>Hit Count</li>
206       * <li>Bytes Free</li>
207       * <li>Optimize Operation Count</li>
208       * <li>Times Optimized</li>
209       * <li>Recycle Count</li>
210       * <li>Recycle Bin Size</li>
211       * <li>Startup Size</li>
212       * </ul>
213       * 
214       * @return cache statistics.
215       */
216      public Map<String, String> getStatistics() {
217        IStats st = this.uriCache.getStatistics();
218        IStatElement[] elements = st.getStatElements();
219        Map<String, String> res = new HashMap<String, String>();
220        res.put("Cache type", st.getTypeName());
221        for (IStatElement e : elements) {
222          res.put(e.getName(), e.getData());
223        }
224        return res;
225      }
