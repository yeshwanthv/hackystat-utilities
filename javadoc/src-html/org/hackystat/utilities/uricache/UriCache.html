001    package org.hackystat.utilities.uricache;
002    
003    import java.util.Properties;
004    
005    import javax.xml.datatype.XMLGregorianCalendar;
006    
007    import org.apache.jcs.JCS;
008    import org.apache.jcs.access.exception.CacheException;
009    import org.apache.jcs.engine.CompositeCacheAttributes;
010    import org.apache.jcs.engine.ElementAttributes;
011    import org.apache.jcs.engine.control.CompositeCache;
012    import org.apache.jcs.engine.control.CompositeCacheManager;
013    
014    /**
015     * Provides an easy caching mechanism which is backed by Apache JCS (Java Caching System). Once
016     * cache configured and initialized, it caches pairs &lt;String, Object&gt;.
017     * 
018     * <br/><br/> Using an UriCache has several advantages: it increases performance as it reduces URL
019     * lookups.
020     * 
021     * @author <a href="mailto:seninp@gmail.com">Pavel Senin<a>
022     */
023    public class UriCache {
024    
025      /** Cache default region name. */
026      private static final String uriCaheRegionName = "UriCache";
027    
028      /** JCS cache handler */
029      private JCS uriCache;
030    
031      /** Cache elements idle time. */
032      private Integer uriCacheIdleTime = 86400;
033    
034      /** Cache default capacity. */
035      private Integer uriCacheMaxObjects = 500;
036    
037      /** JCS cache configuration properties. */
038      private Properties uriCacheProperties = null;
039    
040      /**
041       * Constructor, returns a default instance of UriCache.
042       * 
043       * @throws UriCacheException when unable instantiate a cache.
044       */
045      public UriCache() throws UriCacheException {
046        // setup JCS
047        CompositeCacheManager mgr = CompositeCacheManager.getUnconfiguredInstance();
048        uriCacheProperties = setupProperties();
049        mgr.configure(uriCacheProperties);
050        // get access to bug test cache region
051        try {
052          this.uriCache = JCS.getInstance(uriCaheRegionName);
053          @SuppressWarnings("unused")
054          CompositeCache cache = mgr.getCache(uriCaheRegionName);
055        }
056        catch (CacheException e) {
057          throw new UriCacheException(e.toString());
058        }
059      }
060    
061      /**
062       * Constructor, returns a customized instance of UriCache.
063       * 
064       * @param maxCacheElements specifies the cache capacity.
065       * @param maxLifeSeconds specifies the memory idle time for the cache elements, once this time
066       *        exceeded, cached object will be removed from the cache.
067       * @throws UriCacheException when unable instantiate a cache.
068       */
069      public UriCache(Integer maxCacheElements, Integer maxLifeSeconds) throws UriCacheException {
070    
071        this.uriCacheMaxObjects = maxCacheElements;
072        this.uriCacheIdleTime = maxLifeSeconds;
073    
074        // setup JCS
075        CompositeCacheManager mgr = CompositeCacheManager.getUnconfiguredInstance();
076        uriCacheProperties = setupProperties();
077        mgr.configure(uriCacheProperties);
078    
079        // get access to bug test cache region
080        try {
081          this.uriCache = JCS.getInstance(uriCaheRegionName);
082          @SuppressWarnings("unused")
083          CompositeCache cache = mgr.getCache(uriCaheRegionName);
084        }
085        catch (CacheException e) {
086          throw new UriCacheException(e.toString());
087        }
088      }
089    
090      /**
091       * Place a new object in the cache, associated with uriString. If there is currently an object
092       * associated with the same URI it is replaced.
093       * 
094       * @param uriString identity (URI) of the object to cache.
095       * @param obj The object to cache.
096       * @throws UriCacheException in case of error.
097       */
098      public void cache(String uriString, Object obj) throws UriCacheException {
099        try {
100          this.uriCache.put(uriString, obj);
101        }
102        catch (CacheException e) {
103          throw new UriCacheException(e);
104        }
105      }
106    
107      /**
108       * Puts object into cache and specifies object lifetime within the cache.
109       * 
110       * @param uriString Identity of the object to cache.
111       * @param obj Object to cache.
112       * @param expirationTimeStamp Object expiration timestamp, object will be removed from the cache
113       *        once System time is greater than the timestamp value.
114       * @throws UriCacheException in case of error.
115       */
116      public void cache(String uriString, Object obj, XMLGregorianCalendar expirationTimeStamp)
117          throws UriCacheException {
118    
119        // first of all calculating the life time of this object from now
120        Long currTime = System.currentTimeMillis();
121        Long lifeTime = ((Integer) expirationTimeStamp.getMillisecond()).longValue() - currTime;
122    
123        ElementAttributes attr = new ElementAttributes();
124        attr.setMaxLifeSeconds(lifeTime);
125    
126        try {
127          this.uriCache.put(uriString, obj, attr);
128        }
129        catch (CacheException e) {
130          throw new UriCacheException(e);
131        }
132      }
133    
134      /**
135       * Removes all elements from the cache.
136       * 
137       * @throws UriCacheException in case of error.
138       */
139      public void clear() throws UriCacheException {
140        if (this.uriCache != null) {
141          try {
142            this.uriCache.clear();
143          }
144          catch (CacheException e) {
145            throw new UriCacheException(e);
146          }
147        }
148      }
149    
150      /**
151       * Retrieves object from the cache.
152       * 
153       * @param uriString URI of the object to search for.
154       * @return The cached object or <em>null</em> if not found.
155       */
156      public Object lookup(String uriString) {
157        return this.uriCache.get(uriString);
158      }
159    
160      /**
161       * Removes an Object from the cache.
162       * 
163       * @param uriString Identity of the object to be removed.
164       * @throws UriCacheException in case of error.
165       */
166      public void remove(String uriString) throws UriCacheException {
167        try {
168          this.uriCache.remove(uriString);
169        }
170        catch (CacheException e) {
171          throw new UriCacheException(e.getMessage());
172        }
173      }
174    
175      /**
176       * Sets the cache auto-expire time. Cache will auto expire elements after specified seconds to
177       * reclaim space.
178       * 
179       * @param seconds The new ShrinkerIntervalSeconds value.
180       */
181      public void setMaxMemoryIdleTimeSeconds(int seconds) {
182        CompositeCacheAttributes attr = (CompositeCacheAttributes) this.uriCache.getCacheAttributes();
183        attr.setUseMemoryShrinker(true);
184        attr.setMaxMemoryIdleTimeSeconds(seconds);
185        attr.setShrinkerIntervalSeconds(seconds);
186        this.uriCache.setCacheAttributes(attr);
187      }
188    
189      /**
190       * Constructs configuration properties for the cache instance.
191       * 
192       * @return cache properties.
193       */
194      private Properties setupProperties() {
195        Properties prop = new Properties();
196        // this is JCS required part - configuring default cache properties
197        prop.setProperty("jcs.default", "");
198        prop.setProperty("jcs.default.cacheattributes",
199            "org.apache.jcs.engine.CompositeCacheAttributes");
200        prop.setProperty("jcs.default.cacheattributes.MemoryCacheName",
201            "org.apache.jcs.engine.memory.lru.LRUMemoryCache");
202        prop.setProperty("jcs.default.cacheattributes.MaxObjects", "500");
203    
204        // UriCache region - elements won't go to disk, but should be marked as expired after five
205        // seconds of non-use
206        prop.setProperty("jcs.region.UriCache", "");
207        prop.setProperty("jcs.region.UriCache.cacheattributes.MemoryCacheName",
208            "org.apache.jcs.engine.memory.lru.LRUMemoryCache");
209    
210        prop.setProperty("jcs.region.UriCache.cacheattributes",
211            "org.apache.jcs.engine.CompositeCacheAttributes");
212        prop.setProperty("jcs.region.UriCache.cacheattributes.MaxObjects", this.uriCacheMaxObjects
213            .toString());
214        prop.setProperty("jcs.region.UriCache.cacheattributes.ShrinkerIntervalSeconds", "60");
215        prop.setProperty("jcs.region.UriCache.cacheattributes.UseMemoryShrinker", "true");
216        prop.setProperty("jcs.region.UriCache.cacheattributes.UseDisk", "false");
217        prop.setProperty("jcs.region.UriCache.cacheattributes.UseRemote", "false");
218        prop.setProperty("jcs.region.UriCache.cacheattributes.UseLateral", "false"); 
219    
220        prop.setProperty("jcs.region.UriCache.elementattributes",
221            "org.apache.jcs.engine.ElementAttributes");
222        prop.setProperty("jcs.region.UriCache.elementattributes.IsEternal", "false");
223        prop.setProperty("jcs.region.UriCache.elementattributes.IsLateral", "false");
224        prop.setProperty("jcs.region.UriCache.elementattributes.MaxLifeSeconds", this.uriCacheIdleTime
225            .toString());
226        prop.setProperty("jcs.region.UriCache.elementattributes.IdleTime", this.uriCacheIdleTime
227            .toString());
228        return prop;
229      }
230    
231    }
