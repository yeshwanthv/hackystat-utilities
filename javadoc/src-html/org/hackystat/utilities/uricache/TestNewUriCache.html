001    package org.hackystat.utilities.uricache;
002    
003    import static org.junit.Assert.assertEquals;
004    import static org.junit.Assert.assertNull;
005    
006    import org.junit.Test;
007    
008    /**
009     * Tests the UriCache class.
010     * 
011     * @author Philip Johnson
012     * 
013     */
014    public class TestNewUriCache {
015      
016      private static final String testSubDir = "TestUriCache";
017    
018      /**
019       * Test simple cache put and get.
020       */
021      @Test
022      public void testSimpleCache() {
023        // Create a cache
024        String key = "key";
025        String value = "value";
026        NewUriCache cache = new NewUriCache("TestSimpleCache", testSubDir);
027        cache.put(key, value);
028        assertEquals("Checking simple get", value, cache.get(key));
029        cache.remove(key);
030        assertNull("Checking non-existant get", cache.get(key));
031      }
032      
033      /**
034       * Test simple cache put and get.
035       */
036      @Test
037      public void testDisposeCache() {
038        // Create a cache
039        String key = "key";
040        String value = "value";
041        String cacheName = "TestDisposeCache";
042        NewUriCache cache = new NewUriCache(cacheName, testSubDir);
043        cache.put(key, value);
044        assertEquals("Checking simple get", value, cache.get(key));
045        // now dispose and try again.
046        NewUriCache.dispose(cacheName);
047        cache = new NewUriCache(cacheName, testSubDir);
048        cache.put(key, value);
049        assertEquals("Checking simple get 2", value, cache.get(key));
050      }
051    
052      /**
053       * Test use of disk cache.
054       */
055      @Test
056      public void testDiskCache() {
057        // Create a cache
058        Double maxLifeDays = 1D;
059        Long capacity = 100L;
060        NewUriCache cache = new NewUriCache("TestDiskCache", testSubDir, maxLifeDays, capacity);
061        // Now do a loop and put more than 100 items in it.
062        for (int i = 1; i < 200; i++) {
063          cache.put(i, i);
064        }
065        // Now check to see that we can retrieve all 200 items.
066        for (int i = 1; i < 200; i++) {
067          assertEquals("Checking retrieval", i, cache.get(i));
068        }
069      }
070      
071      /**
072       * Test that we can expire elements from the cache. 
073       * This test no longer works because we've changed the 
074       * @throws Exception If problems occur.
075       */
076      @Test
077      public void testElementExpiration() throws Exception {
078        // Create a cache with maxLife of 1 second and a maximum of 100 in-memory elements.
079        Double maxLifeDays = 1.157e-5D;
080        Long capacity = 100L;
081        NewUriCache cache = new NewUriCache("TestExpiration", testSubDir, maxLifeDays, capacity);
082        // Now do a loop and put 200 items in it.
083        for (int i = 1; i < 200; i++) {
084          cache.put(i, i);
085        }
086        // Add an element that expires in 3 seconds. 
087        cache.put(300, 300, 8.4e-4D);
088        // Now wait for two seconds.
089        Thread.sleep(2000);
090        // Now check to see that all of the items with the default maxLife are gone.
091        for (int i = 1; i < 200; i++) {
092          assertNull("Checking retrieval", cache.get(i));
093        }
094        // And check that our item with the custom maxLife time is still there.
095        assertEquals("Check non-expired element", 300, cache.get(300));
096        // Now wait one more second, enough time for our custom maxLife time to be exceeded.
097        Thread.sleep(1001);
098        // Now see that our element with the custom maxLife time is now gone.
099        assertNull("Check expired element", cache.get(300));
100      }
101    }
