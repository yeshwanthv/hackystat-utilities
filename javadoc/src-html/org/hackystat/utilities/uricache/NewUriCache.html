001    package org.hackystat.utilities.uricache;
002    
003    import java.io.File;
004    import java.io.Serializable;
005    import java.util.ArrayList;
006    import java.util.Properties;
007    import java.util.logging.Level;
008    import java.util.logging.Logger;
009    
010    import org.apache.jcs.access.exception.CacheException;
011    import org.apache.jcs.engine.ElementAttributes;
012    import org.apache.jcs.engine.control.CompositeCacheManager;
013    import org.apache.jcs.JCS;
014    import org.hackystat.utilities.home.HackystatUserHome;
015    import org.hackystat.utilities.logger.HackystatLogger;
016    
017    /**
018     * Provides a wrapper around Apache JCS (Java Caching System) to facilitate Hackystat caching.
019     * This wrapper provides the following:
020     * <ul>
021     * <li> Automatic configuration of an indexed disk cache backing store.
022     * <li> Disk cache is write-through, saving every cached instance. 
023     * <li> Provides a default maximum life for expiring of entries of one day.
024     * <li> Provides a default maximum in-memory cache size of 50,000 instances before using backing
025     * store.
026     * <li> Ensures that all UriCache instances have a unique name and are created only once.
027     * <li> Constructor uses "days" rather than seconds as time unit for maxLife.
028     * <li> put() uses "hours" rather than seconds as time unit for maxLife.
029     * <li> A more convenient API for setting/getting items from the cache and controlling logging.
030     * <li> Logging of exceptions raised by JCS.
031     * <li> Disables JCS logging messages unless the System property 
032     * org.hackystat.utilities.uricache.enableJCSLogging is set.
033     * <li> Convenient packaging mechanism for required jar files to simplify use in Hackystat.
034     * </ul> 
035     * 
036     * Here's an example usage, where we create a separate cache for each user to hold their sensor
037     * data instances as part of the dailyprojectdata service.
038     * <pre>
039     * SensorBaseClient client = new SensorBaseClient(user, host);
040     * NewUriCache cache = new UriCache(user.getEmail(), "dailyprojectdata");
041     *   :
042     * SensorData data = (SensorData)cache.get(uriString);
043     * if (data == null) {
044     *   // Cache doesn't have it, so retrieve from SensorBase and cache locally for next time.
045     *   data = client.getSensorData(uriString);
046     *   cache.put(uriString, data);
047     * }
048     * </pre>
049     * The cache files are in the directory ~/.hackystat/dailyprojectdata/uricache.
050     * Instances expire from the cache after one day, by default.
051     * The maximum number of in-memory instances is 50,000, by default.  
052     * 
053     * @author Philip Johnson
054     * @author Pavel Senin
055     */
056    public class NewUriCache {
057      
058      /** The number of seconds in a day. **/
059      private static final long secondsInADay = 60L * 60L * 24L;
060      /** Maximum life in seconds that an entry stays in the cache. Default is 1 day. */
061      private static final Long defaultMaxLifeSeconds = secondsInADay;
062      /** Maximum number of in-memory instances before sending items to disk. Default is 50,000. */
063      private static final Long defaultCapacity = 50000L;
064      /** The name of this cache, which defines a "region" in JCS terms. */
065      private String cacheName = null;
066      /** The logger used for cache exception logging. */
067      private Logger logger = null;
068      /** Holds a list of already defined caches to help ensure uniqueness. */
069      private static ArrayList<String> cacheNames = new ArrayList<String>();
070      
071      /**
072       * Creates a new UriCache instance with the specified name.
073       * Good for services who want to create a single cache for themselves, such as "dailyprojectdata".
074       * The cacheName must not have already been instantiated, otherwise we raise a runtime exception.
075       * Logger, IdleTime and Capacity will have default values. 
076       * @param cacheName The name of this cache.
077       * @param subDir the .hackystat subdirectory in which the uricache directory holding the backing
078       * store will be created.
079       */
080      public NewUriCache(String cacheName, String subDir) {
081        this(cacheName, subDir, new Double(defaultMaxLifeSeconds), defaultCapacity);
082      }
083      
084      /**
085       * Creates a new UriCache with the specified parameters. 
086       * The cacheName must not have already been instantiated, otherwise we raise a runtime exception.
087       * @param cacheName The name of this UriCache, which will be used as the JCS "region" and also
088       * define the subdirectory in which the index files will live.
089       * @param subDir the .hackystat subdirectory in which the uricache directory holding the backing
090       * store will be created.
091       * @param maxLifeDays The maximum number of days after which items expire from the cache.
092       * @param capacity The number of in-memory instances to store before sending to disk.
093       */
094      public NewUriCache(String cacheName, String subDir, Double maxLifeDays, Long capacity) {
095        // Check to make sure we have not already instantiated a UriCache with this name.
096        if (NewUriCache.cacheNames.contains(cacheName)) {
097          throw new RuntimeException("Error: the cache region name is in use: " + cacheName);
098        }
099        else {
100          NewUriCache.cacheNames.add(cacheName);
101        }
102        this.cacheName = cacheName;
103        this.logger = HackystatLogger.getLogger(cacheName + ".uricache", subDir);
104        if (!System.getProperties().containsKey("org.hackystat.utilities.uricache.enableJCSLogging")) {
105          Logger.getLogger("org.apache.jcs").setLevel(Level.OFF);
106        }
107        CompositeCacheManager ccm = CompositeCacheManager.getUnconfiguredInstance();
108        long maxLifeSeconds = (long)(maxLifeDays * secondsInADay);
109        ccm.configure(initJcsProps(cacheName, subDir, maxLifeSeconds, capacity));
110      }
111      
112      /**
113       * Adds the key-value pair to this cache.
114       * Entry will expire from cache after the default maxLife (currently 24 hours).
115       * Logs a message if the cache throws an exception.
116       * @param key The key, typically a UriString.
117       * @param value The value, typically the object returned from the Hackystat service.
118       */
119      public void put(Serializable key, Serializable value) {
120        try {
121          JCS.getInstance(this.cacheName).put(key, value);
122        }
123        catch (CacheException e) {
124          String msg = "Failure to add " + key + " to cache " + this.cacheName + ":" + e.getMessage();
125          this.logger.warning(msg);
126        }
127      }
128      
129      /**
130       * Adds the key-value pair to this cache.
131       * Logs a message if the cache throws an exception.
132       * @param key The key, typically a UriString.
133       * @param value The value, typically the object returned from the Hackystat service.
134       * @param maxLifeHours The number of hours before this item will expire from cache. 
135       */
136      public void put(Serializable key, Serializable value, double maxLifeHours) {
137        try {
138          ElementAttributes attributes = new ElementAttributes();
139          long maxLifeSeconds = (long)(maxLifeHours * 3600D);
140          attributes.setMaxLifeSeconds(maxLifeSeconds);
141          attributes.setIsEternal(false);
142          JCS.getInstance(this.cacheName).put(key, value, attributes);
143        }
144        catch (CacheException e) {
145          String msg = "Failure to add " + key + " to cache " + this.cacheName + ":" + e.getMessage();
146          this.logger.warning(msg);
147        }
148      }
149      
150      /**
151       * Returns the object associated with key from the cache, or null if not found.
152       * Logs a message if the cache throws an exception.
153       * @param key The key whose associated value is to be retrieved.
154       * @return The value, or null if not found.
155       */
156      public Object get(Serializable key) {
157        try {
158          return JCS.getInstance(this.cacheName).get(key);
159        }
160        catch (CacheException e) {
161          String msg = "Failure of get: " + key + " in cache " + this.cacheName + ":" + e.getMessage();
162          this.logger.warning(msg);
163          return null;
164        }
165      }
166    
167      /**
168       * Ensures that the key-value pair associated with key is no longer in this cache.
169       * Logs a message if the cache throws an exception.
170       * @param key The key to be removed. 
171       */
172      public void remove(Serializable key) {
173        try {
174          JCS.getInstance(this.cacheName).remove(key);
175        }
176        catch (CacheException e) {
177          String msg = "Failure to remove: " + key + " cache " + this.cacheName + ":" + e.getMessage();
178          this.logger.warning(msg);
179        }
180      }
181      
182      /**
183       * Removes everything in this cache.
184       */
185      public void clear() {
186        try {
187          JCS.getInstance(this.cacheName).clear();
188        }
189        catch (CacheException e) {
190          String msg = "Failure to clear cache " + this.cacheName + ":" + e.getMessage();
191          this.logger.warning(msg);
192        }
193      }
194      
195      /**
196       * Shuts down the specified cache, and removes it from the list of active caches so it can be
197       * created again.
198       * @param cacheName The name of the cache to dispose of.
199       */
200      public static void dispose(String cacheName) {
201        try {
202          cacheNames.remove(cacheName);
203          JCS.getInstance(cacheName).dispose();
204        }
205        catch (CacheException e) {
206          String msg = "Failure to clear cache " + cacheName + ":" + e.getMessage();
207          System.out.println(msg);
208        }
209        
210      }
211    
212      /**
213       * Sets up the Properties instance for configuring this JCS cache instance.  Each UriCache is
214       * defined as a JCS "region".  Given a UriCache named "PJ", we create a properties instance
215       * whose contents are similar to the following:
216       * <pre>
217       * jcs.region.PJ=DC-PJ
218       * jcs.region.PJ.cacheattributes=org.apache.jcs.engine.CompositeCacheAttributes
219       * jcs.region.PJ.cacheattributes.MaxObjects=[maxCacheCapacity]
220       * jcs.region.PJ.cacheattributes.MemoryCacheName=org.apache.jcs.engine.memory.lru.LRUMemoryCache
221       * jcs.region.PJ.cacheattributes.UseMemoryShrinker=true
222       * jcs.region.PJ.cacheattributes.MaxMemoryIdleTimeSeconds=3600
223       * jcs.region.PJ.cacheattributes.ShrinkerIntervalSeconds=60
224       * jcs.region.PJ.cacheattributes.MaxSpoolPerRun=500
225       * jcs.region.PJ.elementattributes=org.apache.jcs.engine.ElementAttributes
226       * jcs.region.PJ.elementattributes.IsEternal=false
227       * jcs.region.PJ.elementattributes.MaxLifeSeconds=[maxIdleTime]
228       * jcs.auxiliary.DC-PJ=org.apache.jcs.auxiliary.disk.indexed.IndexedDiskCacheFactory
229       * jcs.auxiliary.DC-PJ.attributes=org.apache.jcs.auxiliary.disk.indexed.IndexedDiskCacheAttributes
230       * jcs.auxiliary.DC-PJ.attributes.DiskPath=[cachePath]
231       * jcs.auxiliary.DC-PJ.attributes.maxKeySize=1000000
232       * </pre>
233       * We define cachePath as HackystatHome.getHome()/.hackystat/[cacheSubDir]/cache.  This enables
234       * a service a cache name of "dailyprojectdata" and have the cache data put inside its
235       * internal subdirectory. 
236       * 
237       * See bottom of: http://jakarta.apache.org/jcs/BasicJCSConfiguration.html for more details.
238       * 
239       * @param cacheName The name of this cache, used to define the region properties.
240       * @param subDir The subdirectory name, used to generate the disk storage directory.
241       * @param maxLifeSeconds The maximum life of instances in the cache in seconds before they expire.
242       * @param maxCapacity The in-memory cache capacity before it goes to disk.
243       * @return The properties file. 
244       */
245      private Properties initJcsProps(String cacheName, String subDir, Long maxLifeSeconds, 
246          Long maxCapacity) {
247        String reg = "jcs.region." + cacheName;
248        String regCacheAtt = reg + ".cacheattributes";
249        String regEleAtt = reg + ".elementattributes";
250        String aux = "jcs.auxiliary.DC-" + cacheName;
251        String auxAtt = aux + ".attributes";
252        String memName = "org.apache.jcs.engine.memory.lru.LRUMemoryCache";
253        String diskAttName = "org.apache.jcs.auxiliary.disk.indexed.IndexedDiskCacheAttributes";
254        Properties props = new Properties();
255        props.setProperty(reg, "DC-" + cacheName);
256        props.setProperty(regCacheAtt, "org.apache.jcs.engine.CompositeCacheAttributes");
257        props.setProperty(regCacheAtt + ".MaxObjects", maxCapacity.toString());
258        props.setProperty(regCacheAtt + ".MemoryCacheName", memName);
259        props.setProperty(regCacheAtt + ".UseMemoryShrinker", "true");
260        props.setProperty(regCacheAtt + ".MaxMemoryIdleTimeSeconds", "3600");
261        props.setProperty(regCacheAtt + ".ShrinkerIntervalSeconds", "60");
262        props.setProperty(regCacheAtt + ".DiskUsagePattern", "UPDATE");
263        props.setProperty(regCacheAtt + ".MaxSpoolPerRun", "500");
264        props.setProperty(regEleAtt, "org.apache.jcs.engine.ElementAttributes");
265        props.setProperty(regEleAtt + ".IsEternal", "false");
266        props.setProperty(regEleAtt + ".MaxLifeSeconds", maxLifeSeconds.toString());
267        props.setProperty(aux, "org.apache.jcs.auxiliary.disk.indexed.IndexedDiskCacheFactory");
268        props.setProperty(auxAtt, diskAttName);
269        props.setProperty(auxAtt + ".DiskPath", getCachePath(subDir));
270        props.setProperty(auxAtt + ".maxKeySize", "100000");
271        return props;
272      }
273      
274      /**
275       * Returns the fully qualified file path to the directory in which the backing store files for
276       * this cache will be placed. Creates the path if it does not already exist.
277    
278       * @param cacheSubDir The subdirectory where we want to locate the cache files.
279       * @return The fully qualified file path to the location where we should put the index files. 
280       */
281      private String getCachePath(String cacheSubDir) {
282        File path = new File(HackystatUserHome.getHome(), ".hackystat/" + cacheSubDir + "/uricache");
283        path.mkdirs();
284        return path.getAbsolutePath();
285      }
