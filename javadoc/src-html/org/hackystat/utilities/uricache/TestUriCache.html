076       */
077      @Test
078      public void testCacheInstance() {
079        try {
080          testCache = new UriCache<String, String>(prop);
081          testCache.cache("1", testString1);
082          testCache.cache("2", testString2);
083          testCache.cache("3", testString3);
084        }
085        catch (UriCacheException e) {
086          fail("Unable to create cache instance: "
087              + formatter.format(new LogRecord(Level.ALL, e.toString())));
088        }
089        assertEquals("Testing .getName() method.", "testCache", testCache.getRegionName());
090        assertEquals("Testing .lookup() method.", testString1, testCache.lookup("1"));
091        assertEquals("Testing .lookup() method.", testString2, testCache.lookup("2"));
092        assertEquals("Testing .lookup() method.", testString3, testCache.lookup("3"));
093      }

095      /**
096       * Cache load test.
097       */
098      @Test
099      public void testCacheLoad() {
100        try {
101          // create a cache instance
102          testCache = new UriCache<String, String>(prop);
103    
104          // put items in cache
105          int cnt = TestUriCache.cacheLoadLimit;
106          for (int i = 0; i < cnt; i++) {
107            testCache.cache("key:" + i, "data:" + i);
108            // System.out.println("cached: " + i);
109          }
110    
111          // get items from cache
112          for (int i = 0; i < cnt; i++) {
113            String element = (String) testCache.lookup("key:" + i);
114            assertNotNull("presave, Should have recevied an element. " + i, element);
115            assertEquals("presave, element is wrong.", "data:" + i, element);
116            // System.out.println("got: " + i);
117          }
118    
119          // Remove all the items
120          for (int i = 0; i <= cnt; i++) {
121            testCache.remove("key:" + i);
122          }
123    
124          // Verify removal
125          for (int i = 0; i <= cnt; i++) {
126            assertNull("Removed key should be null: " + i + ":key", testCache.lookup("key:" + i));
127          }
128    
129        }
130        catch (UriCacheException e) {
131          fail("Unable to proceed with load test: "
132              + formatter.format(new LogRecord(Level.ALL, e.toString())));
133        }
134      }

184      /**
185       * Creates and loads three different caches simultaneously.
186       */
187      @Test
188      public void testCaches() {
189    
190        // create three properties first
191        UriCacheProperties prop1 = new UriCacheProperties();
192        prop1.setCacheStoragePath("sandbox/cache");
193        prop1.setCacheRegionName("StringsCache");
194    
195        UriCacheProperties prop2 = new UriCacheProperties();
196        prop2.setCacheStoragePath("sandbox/cache");
197        prop2.setCacheRegionName("IntegerCache");
198    
199        UriCacheProperties prop3 = new UriCacheProperties();
200        prop3.setCacheStoragePath("sandbox/cache");
201        prop3.setCacheRegionName("DoubleCache");
202    
203        try {
204          // create a cache instances
205          UriCache<String, String> testCache1 = new UriCache<String, String>(prop1);
206          UriCache<String, Integer> testCache2 = new UriCache<String, Integer>(prop2);
207          UriCache<String, Double> testCache3 = new UriCache<String, Double>(prop3);
208    
209          // put items in cache
210          int cnt = TestUriCache.cacheLoadLimit;
211          for (int i = 0; i < cnt; i++) {
212            testCache1.cache("key:" + i, "data:" + i);
213            testCache2.cache("key:" + i, i);
214            testCache3.cache("key:" + i, ((Integer) i).doubleValue());
215          }
216    
217          // get items from cache
218          for (int i = 0; i < cnt; i++) {
219            String element = (String) testCache1.lookup("key:" + i);
220            assertNotNull("presave, Should have recevied an element. " + i, element);
221            assertEquals("presave, element is wrong.", "data:" + i, element);
222            // System.out.println("got: " + i);
223    
224            Integer element1 = (Integer) testCache2.lookup("key:" + i);
225            assertNotNull("presave, Should have recevied an element. " + i, element1);
226            assertEquals("presave, element is wrong.", ((Integer) i), element1);
227    
228            Double element2 = (Double) testCache3.lookup("key:" + i);
229            assertNotNull("presave, Should have recevied an element. " + i, element2);
230            assertEquals("presave, element is wrong.", ((Integer) i).doubleValue(), element2
231                .doubleValue(), 0.02d);
232          }
233    
234          // shutdown caches
235          testCache1.shutdown();
236          testCache2.shutdown();
237          testCache3.shutdown();
238        }
239        catch (UriCacheException e) {
240          fail("Unable to proceed with load test: "
241              + formatter.format(new LogRecord(Level.ALL, e.toString())));
242        }
243    
244      }
