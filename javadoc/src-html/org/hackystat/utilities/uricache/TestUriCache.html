087      /**
088       * Cache persistence test.
089       */
090      @Test
091      public void testCachePersistence() {
092        try {
093          //
094          // get cache instance, clear any leftovers and load cache with new data
095          //
096          testCache = new UriCache<String, String>("testOptimizerCache", prop);
097          testCache.clear();
098          int cnt = 10000;
099          for (int i = 0; i < cnt; i++) {
100            testCache.cache("key:" + i, "data:" + i);
101          }
102          //
103          // now, shut it down
104          //
105          testCache.shutdown();
106          //
107          // chill a little
108          //
109          // System.err.println("UriCache test: testing persistence. Sleeping for 10 seconds.");
110          Thread.yield();
111          Thread.sleep(1000);
112          Thread.yield();
113          //
114          // get cache back alive
115          //
116          testCache = new UriCache<String, String>("testOptimizerCache", prop);
117          //
118          // should be ABLE to read data back
119          //
120          for (int i = 0; i < cnt; i++) {
121            String element = (String) testCache.lookup("key:" + i);
122            assertNotNull("Should have recevied an element. " + i, element);
123            assertEquals("Element is wrong.", "data:" + i, element);
124          }
125        }
126        catch (UriCacheException e) {
127          fail("Unable to create cache instance: "
128              + formatter.format(new LogRecord(Level.ALL, e.toString())));
129        }
130        catch (InterruptedException e) {
131          fail("Unable to sleep >:-!```: " + formatter.format(new LogRecord(Level.ALL, e.toString())));
132        }
133    
134      }

136      /**
137       * Cache shrinker test #1, test shrinking of particular elements.
138       */
139      @Test
140      public void testCacheOptimizer1() {
141        try {
142          //
143          // a little set up.
144          //
145          DatatypeFactory factory = null;
146          factory = DatatypeFactory.newInstance();
147    
148          //
149          // get cache instance, clear any leftovers and load cache with new data
150          //
151          testCache = new UriCache<String, String>("testOptimizerCache", prop);
152          testCache.clear();
153          int cnt = 5000;
154          for (int i = 0; i < cnt; i++) {
155            testCache.cache("key:" + i, "data:" + i);
156          }
157          // put "hot" items in cache now
158          GregorianCalendar calendar = new GregorianCalendar();
159          calendar.setTimeInMillis(System.currentTimeMillis() + 500);
160          for (int i = cnt; i < cnt * 2; i++) {
161            testCache.cache("key:" + i, "data:" + i, factory.newXMLGregorianCalendar(calendar));
162          }
163          //
164          // chill a little, need time to close cache file.
165          //
166          // System.out.println("UriCache test: testing shrinker #1. Sleeping for 10 seconds.");
167          Thread.yield();
168          Thread.sleep(1000);
169          Thread.yield();
170          //
171          // should be unable to read first data block back
172          //
173          for (int i = 0; i < cnt; i++) {
174            String element = (String) testCache.lookup("key:" + i);
175            assertNotNull("Should have recevied an element. " + i, element);
176            assertEquals("Element is wrong.", "data:" + i, element);
177          }
178          //
179          // and should be unable to read second data block back
180          //
181          for (int i = cnt; i < cnt * 2; i++) {
182            testCache.remove("key:" + i);
183            assertNull("Should have NOT recevied an element. " + i, testCache.lookup("key:" + i));
184          }
185        }
186        catch (UriCacheException e) {
187          fail("Unable to create cache instance: "
188              + formatter.format(new LogRecord(Level.ALL, e.toString())));
189        }
190        catch (InterruptedException e) {
191          fail("Unable to sleep >:-!```: " + formatter.format(new LogRecord(Level.ALL, e.toString())));
192        }
193        catch (DatatypeConfigurationException e) {
194          fail("Unable to get DataFactory instance: "
195              + formatter.format(new LogRecord(Level.ALL, e.toString())));
196        }
197    
198      }

241      /**
242       * Tests cache instantiation and simple cache, remove and clean routines.
243       */
244      @Test
245      public void testCacheInstance() {
246        try {
247          //
248          // get cache instance and clear any leftovers
249          //
250          testCache = new UriCache<String, String>("testCache", prop);
251          testCache.clear();
252          //
253          // load cache with new data
254          //
255          int cnt = 10000;
256          for (int i = 0; i < cnt; i++) {
257            testCache.cache("key:" + i, "data:" + i);
258          }
259          //
260          // read data back
261          //
262          for (int i = 0; i < cnt; i++) {
263            String element = (String) testCache.lookup("key:" + i);
264            assertNotNull("Should have recevied an element. " + i, element);
265            assertEquals("Element is wrong.", "data:" + i, element);
266          }
267          //
268          // clean cache one by one and check removal
269          //
270          for (int i = 0; i < cnt; i++) {
271            testCache.remove("key:" + i);
272            assertNull("Should have NOT recevied an element. " + i, testCache.lookup("key:" + i));
273          }
274          //
275          // load cache again
276          //
277          for (int i = 0; i < cnt; i++) {
278            testCache.cache("key:" + i, "data:" + i);
279          }
280          //
281          // clean it
282          //
283          testCache.clear();
284          //
285          // check if it clean
286          //
287          for (int i = 0; i < cnt; i++) {
288            assertNull("Should have NOT recevied an element. " + i, testCache.lookup("key:" + i));
289          }
290        }
291        catch (UriCacheException e) {
292          fail("Unable to create cache instance: "
293              + formatter.format(new LogRecord(Level.ALL, e.toString())));
294        }
