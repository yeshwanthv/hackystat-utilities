001    package org.hackystat.utilities.tstamp;
002    
003    import java.util.GregorianCalendar;
004    import java.util.logging.LogManager;
005    import java.util.logging.Logger;
006    
007    import javax.xml.datatype.DatatypeConfigurationException;
008    import javax.xml.datatype.DatatypeConstants;
009    import javax.xml.datatype.DatatypeFactory;
010    import javax.xml.datatype.XMLGregorianCalendar;
011    
012    /**
013     * Utility class that facilitates Timestamp representation and processing. 
014     * There are too many classes already named "Timestamp", thus the abbreviated name.
015     * @author Philip Johnson
016     */
017    public class Tstamp {
018      
019      /**
020       * Returns true if the passed string can be parsed into an XMLGregorianCalendar object.
021       * @param lexicalRepresentation The string representation.
022       * @return True if the string is a legal XMLGregorianCalendar. 
023       */
024      public static boolean isTimestamp(String lexicalRepresentation) {
025        try {
026          DatatypeFactory factory = DatatypeFactory.newInstance();
027          factory.newXMLGregorianCalendar(lexicalRepresentation);
028          return true;
029          
030        }
031        catch (Exception e) {
032          return false;
033        }
034      }
035      
036      /**
037       * Returns an XMLGregorianCalendar, given its string representation. 
038       * @param lexicalRepresentation The string representation.
039       * @return The timestamp. 
040       * @throws Exception If the string cannot be parsed into a timestamp. 
041       */
042      public static XMLGregorianCalendar makeTimestamp(String lexicalRepresentation) throws Exception {
043        DatatypeFactory factory = DatatypeFactory.newInstance();
044        return factory.newXMLGregorianCalendar(lexicalRepresentation);
045      }
046      
047      /**
048       * Converts a javax.sql.Timestamp into a javax.xml.datatype.XMLGregorianCalendar.
049       * @param tstamp The javax.sql.Timestamp
050       * @return An new instance of a javax.xml.datatype.XmlGregorianCalendar
051       */
052      public static XMLGregorianCalendar makeTimestamp(java.sql.Timestamp tstamp) {
053        DatatypeFactory factory = null;
054        try {
055          factory = DatatypeFactory.newInstance();
056          GregorianCalendar calendar = new GregorianCalendar();
057          calendar.setTimeInMillis(tstamp.getTime());
058          return factory.newXMLGregorianCalendar(calendar);
059        }
060        catch (DatatypeConfigurationException e) {
061          Logger logger = LogManager.getLogManager().getLogger("org.hackystat.sensorbase");
062          logger.warning("Failed to create DatatypeFactory in makeTimestamp()");
063        }
064        return null;
065      }
066      
067      /**
068       * Returns a new java.sql.Timestamp created from a javax.xml.datatype.XMLGregorianCalendar.
069       * @param calendar The XML timestamp.
070       * @return The SQL timestamp.
071       */
072      public static java.sql.Timestamp makeTimestamp(XMLGregorianCalendar calendar) {
073        return new java.sql.Timestamp(calendar.toGregorianCalendar().getTimeInMillis());
074      }
075      
076      /**
077       * Returns an XMLGregorianCalendar corresponding to the current time.
078       * @return The timestamp. 
079       */
080      public static XMLGregorianCalendar makeTimestamp() {
081        try {
082          DatatypeFactory factory = DatatypeFactory.newInstance();
083          return factory.newXMLGregorianCalendar(new GregorianCalendar());
084        }
085        catch (Exception e) {
086          throw new RuntimeException("Bad datatypeFactory", e);
087        }
088      }
089      
090      /**
091       * Returns an XMLGregorianCalendar corresponding to 01-Jan-1000.
092       * @return The timestamp. 
093       */
094      public static XMLGregorianCalendar getDefaultProjectStartTime() {
095        try {
096          DatatypeFactory factory = DatatypeFactory.newInstance();
097          XMLGregorianCalendar startTime = factory.newXMLGregorianCalendar();
098          startTime.setDay(1);
099          startTime.setMonth(1);
100          startTime.setYear(1000);
101          startTime.setTime(0, 0, 0);
102          startTime.setMillisecond(000); //NOPMD
103          return startTime; 
104        }
105        catch (Exception e) {
106          throw new RuntimeException("Bad datatypeFactory", e);
107        }
108      }
109    
110      /**
111       * Returns an XMLGregorianCalendar corresponding to 01-Jan-3000.
112       * @return The timestamp. 
113       */
114      public static XMLGregorianCalendar getDefaultProjectEndTime() {
115        try {
116          DatatypeFactory factory = DatatypeFactory.newInstance();
117          XMLGregorianCalendar endTime = factory.newXMLGregorianCalendar();
118          endTime.setDay(1);
119          endTime.setMonth(1);
120          endTime.setYear(3000);
121          endTime.setTime(23, 59, 59);
122          endTime.setMillisecond(999);      
123          return endTime; 
124        }
125        catch (Exception e) {
126          throw new RuntimeException("Bad datatypeFactory", e);
127        }
128      }  
129      
130      /**
131       * Returns true if tstamp is equal to or between start and end.
132       * @param start The start time.
133       * @param end The end time.
134       * @param tstamp The timestamp to test. 
135       * @return True if between this interval.
136       */  
137      public static boolean inBetween(XMLGregorianCalendar start, XMLGregorianCalendar end, 
138          XMLGregorianCalendar tstamp) {
139        if ((start.compare(tstamp) == DatatypeConstants.EQUAL) ||
140            (end.compare(tstamp) == DatatypeConstants.EQUAL)) {
141          return true;
142        }
143        if ((start.compare(tstamp) == DatatypeConstants.LESSER) &&
144            (end.compare(tstamp) == DatatypeConstants.GREATER)) {
145          return true;
146        }
147        return false;
148      }
149      
150      /**
151       * Returns true if time1 > time2.
152       * @param time1 The first time. 
153       * @param time2 The second time. 
154       * @return True if time1 > time2
155       */
156      public static boolean greaterThan(XMLGregorianCalendar time1, XMLGregorianCalendar time2) {
157        return time1.compare(time2) == DatatypeConstants.GREATER;
158      }
159      
160      /**
161       * Returns true if timeString1 > timeString2.
162       * Throws an unchecked IllegalArgument exception if the strings can't be converted to timestamps.
163       * @param timeString1 The first time. 
164       * @param timeString2 The second time. 
165       * @return True if time1 > time2
166       */
167      public static boolean greaterThan(String timeString1, String timeString2) {
168        try {
169          DatatypeFactory factory = DatatypeFactory.newInstance();
170          XMLGregorianCalendar time1 = factory.newXMLGregorianCalendar(timeString1);
171          XMLGregorianCalendar time2 = factory.newXMLGregorianCalendar(timeString2);
172          return time1.compare(time2) == DatatypeConstants.GREATER;
173        }
174        catch (Exception e) {
175          throw new IllegalArgumentException("Illegal timestring", e);
176        }
177      }
178      
179      /**
180       * Returns true if time1 < time2.
181       * @param time1 The first time. 
182       * @param time2 The second time. 
183       * @return True if time1 < time2
184       */
185      public static boolean lessThan(XMLGregorianCalendar time1, XMLGregorianCalendar time2) {
186        return time1.compare(time2) == DatatypeConstants.LESSER;
187      }
188    
189      /**
190       * Returns true if time1 equals time2
191       * @param time1 The first time. 
192       * @param time2 The second time. 
193       * @return True if time1 equals time2
194       */
195      public static boolean equal(XMLGregorianCalendar time1, XMLGregorianCalendar time2) {
196        return time1.compare(time2) == DatatypeConstants.EQUAL;
197      }
198      
199      
200    }
