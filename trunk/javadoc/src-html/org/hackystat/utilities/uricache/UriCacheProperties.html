001    package org.hackystat.utilities.uricache;
002    
003    import java.util.Properties;
004    import java.util.logging.Level;
005    
006    /**
007     * Provides configuration properties for the UriCache library.
008     * 
009     * @author Pavel Senin.
010     * 
011     */
012    public class UriCacheProperties {
013    
014      /** Cache default region name. */
015      private String uriCaheRegionName = "UriCache";
016    
017      /** Cache elements idle time. 24 hours default time. */
018      private Long maxIdleTime = 86400L;
019    
020      /** Cache default capacity. */
021      private Long maxCacheCapacity = 50000L;
022    
023      /** JCS cache configuration properties. */
024      private Properties cacheProperties = null;
025    
026      /** JCS DC plug-in storage path. */
027      private String dcStoragePath = System.getProperties().getProperty("java.io.tmpdir");
028    
029      /** UriCache logging level */
030      private Level loggerLevel = Level.OFF;
031    
032      /**
033       * Creates new instance and sets default properties.
034       */
035      public UriCacheProperties() {
036        cacheProperties = setupProperties();
037      }
038    
039      /**
040       * Sets maximum idle time for this cache items.
041       * 
042       * @param idleTime the item idle time.
043       */
044      public void setMaxIdleTime(Long idleTime) {
045        this.maxIdleTime = idleTime;
046      }
047    
048      /**
049       * Sets maximum capacity for this cache.
050       * 
051       * @param maxCacheCapacity maximal cache capacity.
052       */
053      public void setMaxMemoryCacpacity(Long maxCacheCapacity) {
054        this.maxCacheCapacity = maxCacheCapacity;
055      }
056    
057      /**
058       * Sets the path to the cache storage folder.
059       * 
060       * @param storagePath the cache storage folder.
061       */
062      public void setCacheStoragePath(String storagePath) {
063        this.dcStoragePath = storagePath;
064        this.cacheProperties.setProperty("jcs.auxiliary.indexedDiskCache.attributes.DiskPath",
065            this.dcStoragePath);
066      }
067    
068      /**
069       * Reports the path to the cache storage folder.
070       * 
071       * @return path to the cache storage folder.
072       */
073      public String getCacheStoragePath() {
074        return this.dcStoragePath;
075      }
076    
077      /**
078       * Constructs configuration properties for the cache instance.
079       * 
080       * @return cache properties.
081       */
082      private Properties setupProperties() {
083    
084        Properties prop = new Properties();
085        //
086        // this is JCS required part - configuring default cache properties
087        //
088        prop.setProperty("jcs.default", "indexedDiskCache");
089        prop.setProperty("jcs.default.cacheattributes",
090            "org.apache.jcs.engine.CompositeCacheAttributes");
091        prop.setProperty("jcs.default.cacheattributes.MaxObjects", this.maxCacheCapacity.toString());
092        prop.setProperty("jcs.default.cacheattributes.MemoryCacheName",
093            "org.apache.jcs.engine.memory.lru.LRUMemoryCache");
094        //
095        // ------- UriCache region cache attributes ---------
096        //
097        prop.setProperty("jcs.region." + this.uriCaheRegionName, "indexedDiskCache");
098        prop.setProperty("jcs.region." + this.uriCaheRegionName + ".cacheattributes",
099            "org.apache.jcs.engine.CompositeCacheAttributes");
100        prop.setProperty("jcs.region." + this.uriCaheRegionName + ".cacheattributes.MaxObjects",
101            this.maxCacheCapacity.toString());
102        prop.setProperty("jcs.region." + this.uriCaheRegionName + ".cacheattributes.MemoryCacheName",
103            "org.apache.jcs.engine.memory.lru.LRUMemoryCache");
104        prop.setProperty("jcs.region." + this.uriCaheRegionName
105            + ".cacheattributes.ShrinkerIntervalSeconds", "300");
106        prop.setProperty("jcs.region." + this.uriCaheRegionName + ".cacheattributes.UseMemoryShrinker",
107            "true");
108        prop.setProperty("jcs.region." + this.uriCaheRegionName + ".cacheattributes.UseDisk", "true");
109        prop
110            .setProperty("jcs.region." + this.uriCaheRegionName + ".cacheattributes.UseRemote", "false");
111        prop.setProperty("jcs.region." + this.uriCaheRegionName + ".cacheattributes.UseLateral",
112            "false");
113        prop.setProperty("jcs.region." + this.uriCaheRegionName + ".cacheattributes.MaxSpoolPerRun",
114            "500");
115        //
116        // -------- elements attributes --------
117        //
118        prop.setProperty("jcs.region." + this.uriCaheRegionName + ".elementattributes",
119            "org.apache.jcs.engine.ElementAttributes");
120        prop.setProperty("jcs.region." + this.uriCaheRegionName + ".elementattributes.IsEternal",
121            "false");
122        prop.setProperty("jcs.region." + this.uriCaheRegionName + ".elementattributes.IsLateral",
123            "false");
124        prop.setProperty("jcs.region." + this.uriCaheRegionName + ".elementattributes.MaxLifeSeconds",
125            this.maxIdleTime.toString());
126        prop.setProperty("jcs.region." + this.uriCaheRegionName + ".elementattributes.IdleTime",
127            this.maxIdleTime.toString());
128        prop.setProperty("jcs.region." + this.uriCaheRegionName + ".elementattributes.isSpool", "true");
129        //
130        // -------- disk cache elements attributes --------
131        //
132        prop.setProperty("jcs.auxiliary.indexedDiskCache",
133            "org.apache.jcs.auxiliary.disk.indexed.IndexedDiskCacheFactory");
134        prop.setProperty("jcs.auxiliary.indexedDiskCache.attributes",
135            "org.apache.jcs.auxiliary.disk.indexed.IndexedDiskCacheAttributes");
136        prop.setProperty("jcs.auxiliary.indexedDiskCache.attributes.MaxPurgatorySize", "300000");
137        prop.setProperty("jcs.auxiliary.indexedDiskCache.attributes.MaxKeySize", "500000");
138        prop.setProperty("jcs.auxiliary.indexedDiskCache.attributes.MaxRecycleBinSize", "50000");
139        prop.setProperty("jcs.auxiliary.indexedDiskCache.attributes.OptimizeAtRemoveCount", "300000");
140        prop.setProperty("jcs.auxiliary.indexedDiskCache.attributes.EventQueueType", "SINGLE");
141        prop.setProperty("jcs.auxiliary.indexedDiskCache.attributes.EventQueuePoolName",
142            "disk_cache_event_queue");
143        //
144        // --------- disk cache tricks
145        //
146        prop.setProperty("thread_pool.default.boundarySize", "2000");
147        prop.setProperty("thread_pool.default.maximumPoolSize", "150");
148        prop.setProperty("thread_pool.default.minimumPoolSize", "4");
149        prop.setProperty("thread_pool.default.keepAliveTime", "350000");
150        prop.setProperty("thread_pool.default.whenBlockedPolicy", "RUN");
151        prop.setProperty("thread_pool.default.startUpSize", "4");
152        prop.setProperty("thread_pool.disk_cache_event_queue.useBoundary", "false");
153        prop.setProperty("thread_pool.disk_cache_event_queue.minimumPoolSize", "2");
154        prop.setProperty("thread_pool.disk_cache_event_queue.keepAliveTime", "3500");
155        prop.setProperty("thread_pool.disk_cache_event_queue.startUpSize", "10");
156    
157        return prop;
158    
159      }
160    
161      /**
162       * Reports full set of configuration properties.
163       * 
164       * @return configuration properties.
165       */
166      public Properties getProperties() {
167        return this.cacheProperties;
168      }
169    
170      /**
171       * Reports the cache name associated with this properties object.
172       * 
173       * @return this cache region name.
174       */
175      public String getCacheRegionName() {
176        return this.uriCaheRegionName;
177      }
178    
179      /**
180       * Sets the cache region name.
181       * 
182       * @param cacheName name to set.
183       */
184      public void setCacheRegionName(String cacheName) {
185        this.uriCaheRegionName = cacheName;
186      }
187    
188      /**
189       * Reports the logger level set in the properties.
190       * 
191       * @return logger level.
192       */
193      public Level getLoggerLevel() {
194        return this.loggerLevel;
195      }
196    
197      /**
198       * Set the logger level.
199       * 
200       * @param level logger level to set.
201       */
202      public void setLoggerLevel(Level level) {
203        this.loggerLevel = level;
204      }
205    
206    }
