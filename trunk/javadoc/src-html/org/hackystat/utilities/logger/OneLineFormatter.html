001    package org.hackystat.utilities.logger;
002    
003    import java.text.SimpleDateFormat;
004    import java.util.Date;
005    import java.util.Locale;
006    import java.util.logging.Formatter;
007    import java.util.logging.LogRecord;
008    
009      /**
010       * Provides a one line formatter for use with Hackystat logging. Supports optional date stamp
011       * prefix. If the date stamp prefix is enabled, then a cr is also added.
012       *
013       * @author Philip Johnson
014       */
015    class OneLineFormatter extends Formatter {
016      
017      /**
018       * Default constructor that does nothing. 
019       */
020      public OneLineFormatter () {
021        // Do nothing. 
022      }
023    
024        /**
025         * Formats the passed log string as a single line. Prefixes the log string with a date stamp.
026         *
027         * @param record  A log record.
028         * @return The message string.
029         */
030      @Override
031        public String format(LogRecord record) {
032          SimpleDateFormat dateFormat = new SimpleDateFormat("MM/dd HH:mm:ss", Locale.US);
033          StringBuffer buff = new StringBuffer();
034          buff.append(dateFormat.format(new Date()));
035          buff.append("  ");
036          buff.append(record.getMessage());
037          buff.append(System.getProperty("line.separator"));
038          return buff.toString();
039        }
040      }
